134 void TCPSession::update_timeout()
135 {
136  if(rtt_smoothed)
137  {
138  int delta = rtt_measured - (rtt_smoothed >> TCP_DEFAULT_RTT_SHIFT);
139  if((rtt_smoothed += delta) <= 0) rtt_smoothed = 1;
140  if(delta < 0) delta = -delta;
141  delta -= (rtt_var >> TCP_DEFAULT_RTTVAR_SHIFT);
142  if((rtt_var += delta) <= 0) rtt_var = 1;
143  }
144  else
145  {
146  rtt_smoothed = (rtt_measured + 1) << TCP_DEFAULT_RTT_SHIFT;
147  rtt_var = (rtt_measured + 1) << (TCP_DEFAULT_RTTVAR_SHIFT - 1);
148  }
149 
150  rxmit_timeout_double = mymax(mymin(tcp_master->getSlowTimeoutDouble() * ((rtt_smoothed >> TCP_DEFAULT_RTT_SHIFT) + rtt_var),
151  TCP_DEFAULT_RXMIT_MAX_TIMEOUT), TCP_DEFAULT_RXMIT_MIN_TIMEOUT);
152  rxmit_timeout = tcp_master->inHost()->d2t(rxmit_timeout_double, 0);
