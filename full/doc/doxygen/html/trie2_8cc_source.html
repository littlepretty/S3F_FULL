71  if(nBits > TRIE_KEY_SIZE(key)) { // the given mask is too long
72  fprintf(stderr, "Invalid mask size %d\n",nBits);
73  nBits = TRIE_KEY_SIZE(key);
74  }

76  #ifdef TRIE2_DEBUG
77  RouteInfo* data_ri = (RouteInfo*)data;
78  printf("new route: %s/%d\n",IPPrefix::ip2txt(data_ri->destination.addr),data_ri->destination.len);
79  #endif
80 
81  TrieNode* cur = root;
82  TrieNode* lastcur;
83  int i;
84  int index;
85  for(i=0;cur;i++) {
86  // if we find a match
87  if(cur->data) {
88  // TODO: Tests indicate that true duplicates are still not always detected
89  // if this node already contains the correct forwarding data
90  if(data->equiv(cur->data)) {
91  delete data; // remove duplicate data (doesn't matter which since identical)
92  return 0;
93  // if inserting to this exact point but data is different, we have a real conflict
94  } else if(nBits == i) {
95  #ifdef TRIE2_DEBUG
96  printf("CONFLICT!!!\n");
97  RouteInfo* cur_ri = (RouteInfo*)cur->data;
98  printf("old route: %s/%d\n",IPPrefix::ip2txt(cur_ri->destination.addr),cur_ri->destination.len);
99  printf("old next_hop: %s\n",IPPrefix::ip2txt(cur_ri->next_hop));
100  printf("new next_hop: %s\n",IPPrefix::ip2txt(data_ri->next_hop));
101  #endif
102  if(!replace) {
103  return data;
104  }
105 
106  TrieData* temp = cur->data;
107  cur->data = data;
108  return temp;
109  }
110  }
111 
112  // check next node
113  key <<= 1;
114  lastcur = cur;
115  index = (key & TRIE_BIT_MASK)?1:0;
116  cur = cur->children[index];
117  }
118 
119  #ifdef TRIE2_DEBUG
120  printf("Stored at depth %d\n",i); // DEBUG
121  #endif
122 
123  // cur is null, so lastcur is where we need to insert
124  lastcur->children[index] = new TrieNode();

130 /*
131  * Performs a fast longest-matching prefix search for the key. The
132  * reason to do so is that it's used in routing entry searches.
133  * Because it's so critical to routing performance, a fast search is
134  * necessary.
135  *
136  * We assume the search length is independent. that means we try to
137  * find the best.
138  */
139 TrieData* Trie2::lookup(TRIEBITSTRING key)
140 {
141  // If this Trie exists to call 'lookup', root has already been asserted as non-null
142  TrieNode* cur = root->children[(key & TRIE_BIT_MASK)?1:0];
143  TrieData* pData = root->data; // handles default-route case
144 
145  while(cur) {
146  if(cur->data) {
147  pData = cur->data;
148  }
149  key <<= 1; // needed from pre-loop lookup and removes last unnecessary shift
150  cur = cur->children[(key & TRIE_BIT_MASK)?1:0];
151  }
152 
153  return pData;
154 }

156 /*
157  * Removes the key from the trie, if it exists, and returns the data
158  * corresponding to the removed key. If the key is not in the trie,
159  * return 0.
160  * In this Trie implementation, remove should be used with reservation as for
161  * nBits < TRIE_KEY_SIZE(key), multiple routes may actually be removed.
162  */
163 TrieData* Trie2::remove(TRIEBITSTRING key, int nBits) {
164  if(nBits > TRIE_KEY_SIZE(key)) { // the given mask is too long
165  fprintf(stderr, "Invalid mask size %d\n",nBits);
166  nBits = TRIE_KEY_SIZE(key);
167  }
168 
169  TrieNode* cur = root;
170  TrieNode* lastcur;
171  int i;
172  int index;
173  // move to target depth
174  for(i=0;cur && i < nBits;i++) {
175  key <<= 1;
176  lastcur = cur;
177  index = (key & TRIE_BIT_MASK)?1:0;
178  cur = cur->children[index];
179  }
180 
181  // if there is a match at the proper depth
182  if(cur && cur->data) {
183  TrieData* olddata = cur->data;
184  cur->data = 0;
185  // free this node if it is now a leaf
186  if(cur->children[0] == NULL && cur->children[1] == NULL) {
187  delete lastcur->children[index];
188  }
189 
190  return olddata;
191  }
192 
193  // no match was found at the given depth
194  return 0;
195 }

319 void Trie2::iterator::nextNode() {
320  bool keepMoving = true;
321  while(keepMoving) {
322  NodeNext* nn = stateStack.top();
323  stateStack.pop();
324  currentNode = nn->node;
325  assert(currentNode);
326  switch (nn->next) {
327  case LEFT: {
328  stateStack.push(new NodeNext(nn->node, RIGHT)); // afterwards move right
329  TrieNode* leftChild = currentNode->children[LEFT];
330  if (leftChild) { // has left child?
331  stateStack.push(new NodeNext(leftChild, LEFT)); // descend left
332  TRIE2_DUMP(std::cout << currentNode << " push RIGHT, LEFT down, push LEFT (" <<
333  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
334  } else {
335  TRIE2_DUMP(std::cout << currentNode << " push RIGHT (" << stateStack.size() <<
336  ") - "; currentNode->dump(); std::cout << std::endl);
337  }
338  }
339  break;
340  case RIGHT: {
341  stateStack.push(new NodeNext(nn->node, UP)); // afterwards move up
342  TrieNode* rightChild = currentNode->children[RIGHT];
343  if (rightChild) { // has right child?
344  TRIE2_DUMP(std::cout << currentNode << " push UP, RIGHT down, push LEFT (" <<
345  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
346  stateStack.push(new NodeNext(rightChild, LEFT)); // afterwards move left
347  } else {
348  TRIE2_DUMP(std::cout << currentNode << " push UP (" << stateStack.size() <<
349  ") - "; currentNode->dump(); std::cout << std::endl);
350  }
351  }
352  break;
353  case UP:
354  // visit node
355  keepMoving = false;
356  TRIE2_DUMP(std::cout << currentNode << " visit (" << stateStack.size() <<
357  ") - "; currentNode->dump(); std::cout << std::endl);
358  assert(currentNode != myTrie->root || stateStack.size() == 0);
359  break;
360  default:
361  // ERROR: undefined state of trie iterator
362  throw State();
363  } // switch
364  delete nn;
365  } // while
366 }
