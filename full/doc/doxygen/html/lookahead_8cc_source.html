114 /* predict lookahead amount */
115 void LookaheadANN::post_cycle(ltime_t barrier)
116 {
117  /* step 1: record send packet into send buffer */
118  record_packet(ve->sendq, barrier, true);
119 
120  bool historychanged = false;
121 
122  /* step 2: process new send/recv packets */
123  while (!sendbuf.empty() || !recvbuf.empty())
124  {
125  /* find the packet with smallest timestamp */
126  bool bsend;
127  if (sendbuf.empty()) bsend = false;
128  else if (recvbuf.empty()) bsend = true;
129  else if (sendbuf.front()->timestamp > recvbuf.front()->timestamp) bsend = false;
130  else bsend = true;
131 
132  point *pt;
133  if (bsend) {
134  pt = sendbuf.front();
135  sendbuf.pop_front();
136  }
137  else {
138  pt = recvbuf.front();
139  recvbuf.pop_front();
140  }
141 
142  /* check for old history points expiration */
143  if (!history.empty() && pt->timestamp - history.back()->timestamp > inputfilter)
144  history.clear();
145 
146  /* form a new lookahead entry if have enough history data */
147  if (history.size() == npoint + 1) {
148  entry *ent = new entry;
149  entry_prepare(ent);
150  ent->output[0] = -3.0 + log10(
151  max((ltime_t)1, pt->timestamp-history.back()->timestamp));
152 
153  /* send the new entry for training or validation */
154  if (ann) entry_valid(ent, bsend);
155  else entry_train(ent);
156  }
157 
158  /* push the packet into history */
159  history.push_back(pt);
160  historychanged = true;
161  if (history.size() > npoint + 1) {
162  pt = history.front();
163  history.pop_front();
164  delete pt;
165  }
166  }
167 
168  /* step 3: predit lookahead amount */
169  ltime_t advance = history.empty() ? inputfilter + 1 :
170  ve->get_clock() - history.back()->timestamp;
171  if (advance > inputfilter) {
172  ve->lookahead = outputdefault;
173  }
174  else if (!ann || history.size() < npoint + 1) {
175  ve->lookahead = 0;
176  }
177  else {
178  if (historychanged) {
179  entry *ent = new entry;
180  entry_prepare(ent);
181  lastpredict = entry_predict(ent);
182  }
183  ve->lookahead = max( (ltime_t)0, lastpredict - advance );
184  ve->lookahead = min( (ltime_t)outputfilter, ve->lookahead );
185  }
186 
187  /* step 4: check VE idle information for potential larger lookahead */
188  if (ann && ve->idle) {
189  ltime_t firstevent = -1;
190  if (!ve->timerq.empty()) firstevent = ve->timerq.top().due;
191  if (!ve->recvq.empty()) {
192  ltime_t temp = ve->recvq.front()->timestamp;
193  if (firstevent < 0 || temp < firstevent) firstevent = temp;
194  }
195  ltime_t idlela = firstevent < 0 ? outputdefault : firstevent - ve->get_clock();
196  ve->lookahead = max( ve->lookahead, idlela );
197  }
198 }

221 void LookaheadANN::record_packet(deque<EmuPacket*> const &pktq, ltime_t barrier, bool send)
222 {
223  for (size_t i = 0; i < pktq.size(); i++)
224  {
225  EmuPacket *pkt = pktq[i];
226  if (pkt->timestamp > barrier) break;
227  point *pt = new point;
228  pt->timestamp = pkt->timestamp;
229 
230  /* extract features from packet */
231  pt->feature[0] = send ? log10(pkt->len) : -log10(pkt->len);
232 
233  unsigned char *packet = pkt->data;
234  sniff_ip *ip = (sniff_ip*)(packet + SIZE_ETHERNET);
235  if (ip->ip_p != 6) { // not a TCP packet
236  pt->feature[1] = -1;
237  pt->feature[2] = -1;
238  }
239  else {
240  int size_ip = IP_HL(ip) * 4; // TCP FIN/SYN flags
241  sniff_tcp *tcp = (sniff_tcp*)(packet + SIZE_ETHERNET + size_ip);
242  pt->feature[1] = (tcp->th_flags & TH_FIN) != 0;
243  pt->feature[2] = (tcp->th_flags & TH_SYN) != 0;
244  }
245 
246  /* push the packet into the buffer */
247  if (send) sendbuf.push_back(pt);
248  else recvbuf.push_back(pt);
249  }
250 }

285

288  ltime_t predict = (ltime_t)exp10(output[0] + 3.0);

290  ltime_t diff = abs(predict - correct);
291 
292  int wrong = 1;
293  if (!send && predict > correct) wrong = 0;
294  else if (1) {
295  if (diff < VTIME_VTIMESLICE) wrong = 0;
296  else if (diff < correct / 10) wrong = 0;
297  }
298  double error = 1.0 * diff / correct;
299  if (diff < VTIME_VTIMESLICE) error = 0.0;
300  if (error > 1.0) error = 1.0;
301

303  predicterr = errmomentum * predicterr + (1-errmomentum) * error;
304 #ifdef SLAVE_LOG
305  if(pflog) fprintf(pflog, "%ld.%06ld\tve%d\tlook\t%ld\t%ld\n",
306  ve->get_clock()/1000000, ve->get_clock()%1000000,
307  ve->get_id(), correct, predict);
308 #endif
309  if (predicterr > retrainerr) {

311  fann_destroy(ann);

343 void LookaheadANN::ann_train()

362  entry *ent = trainset[i];
363  for (int j = 0; j < nanninput; j++) fprintf(pf, "%f ", ent->input[j]);
364  fprintf(pf, "\n");
365  for (int j = 0; j < nannoutput; j++) fprintf(pf, "%f ", ent->output[j]);
366  fprintf(pf, "\n");
367  }
368  fclose(pf);
369 
370 
371  /* read ANN trainset from file */
372  struct fann_train_data *train_data;
373  train_data = fann_read_train_from_file(fn);
374  nentry = train_data->num_data;
375 
376  /* split data into train set and validation set */
377  fann_shuffle_train_data(train_data);
378  int ntrain = (int)(nentry * trainset_percent);
379  struct fann_train_data *valid_data, *temp;
380  valid_data = fann_subset_train_data(train_data, ntrain, nentry-ntrain);
381  temp = fann_subset_train_data(train_data, 0, ntrain);
382  fann_destroy_train(train_data);

384

387  fann_set_training_algorithm(ann, FANN_TRAIN_RPROP);

389  fann_set_activation_function_output(ann, FANN_LINEAR);
390  fann_init_weights(ann, train_data);
391 
392  ann_filename(fn, "ann");
393  double minvalidmse = 1e308;
394  int failcount = 0;
395 
396  /* ANN training */
397  printf("VE%d: training ANN\n", ve->get_id());
398  for (int epoch = 1; epoch <= maxtrainepoch && failcount < valid_fail_count; epoch++)
399  {
400  double trainmse = fann_train_epoch(ann, train_data);
401  double validmse = fann_test_data(ann, valid_data);
402  if (validmse < minvalidmse - 1e-10) {
403  failcount = 0;
404  minvalidmse = validmse;
405  fann_save(ann, fn);
406  }
407  else {
408  failcount++;
409  }
410  if (epoch % print_interval == 0 || failcount == valid_fail_count)
411  printf("epoch=%d trainmse=%.6f validmse=%.6f bestmse=%.6f\n",
412  epoch, trainmse, validmse, minvalidmse);
413  }
414  printf("VE%d: training ANN done\n", ve->get_id());
415 
416  /* read the saved (best) ANN from file */
417  fann_destroy(ann);
418  ann = fann_create_from_file(fn);
419 
420  /* training done */
421  fann_reset_MSE(ann);
422  predicterr = 0.0;
423  deque_clear<entry*>(trainset);
424  assert(trainset.size() == 0);
425  fann_destroy_train(train_data);
426  fann_destroy_train(valid_data);
427 }
