51  /* When a continuation returns successfully, this function will be
52  called. Based on the current status of this continuation, we will
53  decide what to do next. */
54  virtual void success() {
55  UDPServerSession* server = (UDPServerSession*)owner;
56  switch(status) {
57  case UDP_SERVER_SESSION_RECEIVING:
58  server->request_received(this);
59  break;
60  case UDP_SERVER_SESSION_CONNECTING:
61  server->client_connected(this);
62  break;
63  case UDP_SERVER_SESSION_SENDING:
64  server->data_sent(this);
65  break;
66  case UDP_SERVER_SESSION_CLOSING:
67  server->session_closed(this);
68  delete this; // end of the journey!!!
69  break;
70  default:
71  assert(0);
72  break;
73  }
74  }

76  /* When a continuation returns unsuccessfully, this function will be
77  called to print out the error. */
78  virtual void failure() {
79  UDPServerSession* server = (UDPServerSession*)owner;
80  switch(status) {
81  case UDP_SERVER_SESSION_RECEIVING:
82  UDP_DUMP(printf("UDPServerSessionContinuation::failure() on host \"%s\": "
83  "failed to receive request from client.\n",
84  server->inHost()->nhi.toString()));
85  server->handle_client(server_socket);
86  break;
87  case UDP_SERVER_SESSION_CONNECTING:
88  UDP_DUMP(printf("UDPServerSessionContinuation::failure() on host \"%s\": "
89  "failed to connect to client.\n",
90  server->inHost()->nhi.toString()));
91  // since this session has failed, we should decrease the number
92  // of clients in the system.
93  if(server->client_limit && server->nclients-- == server->client_limit)
94  server->handle_client(server_socket);
95  break;
96  case UDP_SERVER_SESSION_SENDING:
97  UDP_DUMP(printf("UDPServerSessionContinuation::failure() on host \"%s\": "

104  break;
105  case UDP_SERVER_SESSION_CLOSING:

283 void UDPServerSession::request_received(UDPServerSessionContinuation* const cnt)
284 {
285  // assume the request can be received with one call
286  assert(cnt->retval == (int)request_size);
287 
288  cnt->file_size = ntohl(*(uint32*)cnt->reqbuf);
289  UDP_DUMP(printf("[host=\"%s\"] %s: session_proc(), socket %d received request "
290  "(request_size=%u, file_size=%u).\n",
291  inHost()->nhi.toString(), getNowWithThousandSeparator(),
292  cnt->server_socket, request_size, cnt->file_size));
293 
294  if(cnt->file_size <= 0)
295  {
296  handle_client(cnt->server_socket);
297  return;
298  }
299 
300  // A side-effect of this 'connected' call is to give remote ip
301  // address and port number
302  sm->connected(cnt->server_socket, &cnt->client_ip, &cnt->client_port);
303 
304  if(!client_limit || nclients++ < client_limit)
305  handle_client(cnt->server_socket);
306 
307  cnt->client_socket = sm->socket();
308  if(!sm->bind(cnt->client_socket, server_ip, ++server_port, UDP_PROTOCOL_NAME, 0))
309  {
310  UDP_DUMP(printf("request_received() on host \"%s\": failed to bind.\n", inHost()->nhi.toString()));
311  if(client_limit && nclients-- == client_limit)
312  handle_client(cnt->server_socket);
313  return;
314  }
315 
316  UDP_DUMP(printf("[host=\"%s\"] %s: session_proc(), socket %d connecting to client: \"%s:%d\".\n",
317  inHost()->nhi.toString(), getNowWithThousandSeparator(),
318  cnt->client_socket, IPPrefix::ip2txt(cnt->client_ip), cnt->client_port));
319 
320  cnt->sent_bytes = 0;
321 
322  cnt->status = UDPServerSessionContinuation::UDP_SERVER_SESSION_CONNECTING;
323  sm->connect(cnt->client_socket, cnt->client_ip, cnt->client_port, cnt);
324 }
