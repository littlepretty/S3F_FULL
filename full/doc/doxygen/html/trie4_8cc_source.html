104  * Inserts a new (key, data) pair.

106 TrieData* Trie4::insert(TRIEBITSTRING key, int nBits,
107  TrieData* data, bool replace) {
108  TrieNode *fail = 0;
109  int bitpos = 0;
110 
111  TrieNode* match = search(key, nBits, &fail, &bitpos);
112 
113  // Does the key already exist in our trie?
114  // If so, replace the data if replace flag is true.
115  if(match) {
116  if(!replace) return data;
117  TrieData* temp = mData->removeElement(match->index,true);
118  match->index = mData->addElement(data);
119  return temp;
120  }
121 
122  // Create all the necessary nodes for the unmatched bits
123  // after fail and insert the data.
124  create(fail, REMOVE(bitpos, key), nBits-bitpos, data);
125  nElements++;
126 
127  return 0;
128 }

130 /*
131  * Performs a fast longest-matching prefix search for the key. The
132  * reason to do so is that it's used in routing entry searches.
133  * Because it's so critical to routing performance, a fast search is
134  * necessary.
135  *
136  * We assume the search length is independent. that means we try to
137  * find the best.
138  */
139 TrieData* Trie4::lookup(TRIEBITSTRING key)
140 {
141  // If this Trie exists to call 'lookup', root has already been asserted as non-null
142  TrieNode* cur = root->children[(key & TRIE_BIT_MASK)?1:0];
143  TrieData* pData;
144  if(root->index != UINT_MAX) {
145  pData = mData->getElement(root->index); // handles default-route case
146  } else {
147  pData = 0;
148  }
149 
150  while(cur) {
151  if(cur->index != UINT_MAX) {
152  pData = mData->getElement(cur->index);
153  }
154  key <<= 1;
155  cur = cur->children[(key & TRIE_BIT_MASK)?1:0];
156  }
157 
158  return pData;
159 }

191 /*
192  * Helper function for Insert and Lookup. This function will search
193  * the trie for the key. If found, it will return a pointer to the
194  * TrieNode, set failedNode to its parent, and set failedbitpos to the
195  * index in the parent's array. If not found, it will return NULL, and
196  * set failedNode to the TrieNode where the search fails and
197  * failedbitpos to the bit position that failed.
198  */
199 TrieNode* Trie4::search(TRIEBITSTRING key, int len,
200  TrieNode** failedNode, int* failedbitpos)
201 {
202  // TODO: Optimize this function
203  TrieNode* cur, *lastcur, *parent;
204  TrieNode* match = 0;
205  int bitpos, curbit, parentbit;
206  if(len > TRIE_KEY_SIZE(key)) {
207  len = TRIE_KEY_SIZE(key);
208  }
209 
210  // Go through the bits starting from the MSB and traverse
211  // the tree. Remember the last match.
212  for(bitpos = 0, lastcur = cur = root;
213  cur && bitpos < len;
214  cur = cur->children[curbit], bitpos++) {
215  if(cur->index != UINT_MAX) {
216  parent = lastcur;
217  parentbit = GETBIT(bitpos-1,key);
218  match = cur;
219  }
220 
221  lastcur = cur;
222  curbit = GETBIT(bitpos, key);
223  }
224 
225  if(cur && cur->index != UINT_MAX) {
226  match = cur;
227  if(failedNode && failedbitpos) {
228  // we have a match on the last node, parent is lastcur
229  *failedNode = lastcur;
230  *failedbitpos = GETBIT(bitpos-1,key);
231  }
232  } else if(failedNode && failedbitpos) {
233  // if there is a match
234  if(match) {
235  *failedNode = parent;
236  *failedbitpos = parentbit;
237  } else { // no match
238  if(cur) {
239  *failedbitpos = bitpos;
240  *failedNode = cur;
241  } else {
242  *failedbitpos = bitpos - 1;
243  *failedNode = lastcur;
244  }
245  }
246  }
247 
248  return match;
249 }

386 void Trie4::iterator::nextNode() {
387  bool keepMoving = true;
388  while(keepMoving) {
389  NodeNext* nn = stateStack.top();
390  stateStack.pop();
391  currentNode = nn->node;
392  assert(currentNode);
393  switch (nn->next) {
394  case LEFT: {
395  stateStack.push(new NodeNext(nn->node, RIGHT)); // afterwards move right
396  TrieNode* leftChild = currentNode->children[LEFT];
397  if (leftChild) { // has left child?
398  stateStack.push(new NodeNext(leftChild, LEFT)); // descend left
399  TRIE4_DUMP(std::cout << currentNode << " push RIGHT, LEFT down, push LEFT (" <<
400  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
401  } else {
402  TRIE4_DUMP(std::cout << currentNode << " push RIGHT (" << stateStack.size() <<
403  ") - "; currentNode->dump(); std::cout << std::endl);
404  }
405  }
406  break;
407  case RIGHT: {
408  stateStack.push(new NodeNext(nn->node, UP)); // afterwards move up
409  TrieNode* rightChild = currentNode->children[RIGHT];
410  if (rightChild) { // has right child?
411  TRIE4_DUMP(std::cout << currentNode << " push UP, RIGHT down, push LEFT (" <<
412  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
413  stateStack.push(new NodeNext(rightChild, LEFT)); // afterwards move left
414  } else {
415  TRIE4_DUMP(std::cout << currentNode << " push UP (" << stateStack.size() <<
416  ") - "; currentNode->dump(); std::cout << std::endl);
417  }
418  }
419  break;
420  case UP:
421  // visit node
422  keepMoving = false;
423  TRIE4_DUMP(std::cout << currentNode << " visit (" << stateStack.size() <<
424  ") - "; currentNode->dump(); std::cout << std::endl);
425  assert(currentNode != myTrie->root || stateStack.size() == 0);
426  break;
427  default:
428  // ERROR: undefined state of trie iterator
429  throw State();
430  } // switch
431  delete nn;
432  } // while
433 }
