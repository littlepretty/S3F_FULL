53 uint32 TCPRecvWindow::generate(uint32 length, byte* msg)
54 {
55  // we always grab one block from the buffer and report it to the
56  // upper layer.
57  uint32 request_len = length;
58  while(seg_buffer && length > 0 && (expect() == seg_buffer->seqno))
59  {
60  // we consider the situation in which the upper layer has
61  // requested different data lengths.
62  if(length >= seg_buffer->length)
63  {
64  // can submit whole data block to the upper layer.
65  if(msg)
66  {
67  if(seg_buffer->msg)
68  {
69  memcpy(msg, seg_buffer->msg, seg_buffer->length);
70  delete[] seg_buffer->msg;
71  }
72  else
73  {
74  memset(msg, 0, seg_buffer->length);
75  }
76  msg += seg_buffer->length;
77  }
78  else if(seg_buffer->msg)
79  {
80  delete[] seg_buffer->msg;
81  }
82 
83  // update the start sequence number
84  shift(seg_buffer->length);
85  length -= seg_buffer->length;
86 
87  TCPSegment* seg = seg_buffer->next;
88  add_a_free_segment(seg_buffer);
89  seg_buffer = seg;
90  }
91  else
92  {
93  // we have to cut the current data block into two pieces.
94  if(msg)
95  {
96  if(seg_buffer->msg)
97  {
98  memcpy(msg, seg_buffer->msg, length);
99  byte* remain_msg = new byte[seg_buffer->length - length];
100  memcpy(remain_msg, seg_buffer->msg+length, seg_buffer->length - length);
101  delete [] seg_buffer->msg;
102  seg_buffer->msg = remain_msg;
103  }
104  else
105  {
106  memset(msg, 0, length);
107  }
108  msg += seg_buffer->length;
109  }
110  else if(seg_buffer->msg)
111  {
112  byte* remain_msg = new byte[seg_buffer->length - length];
113  memcpy(remain_msg, seg_buffer->msg+length, seg_buffer->length - length);
114  delete [] seg_buffer->msg;
115  seg_buffer->msg = remain_msg;
116  }
117 
118  // shorten the length of the remaining message.
119  seg_buffer->seqno += length;
120  seg_buffer->length -= length;
121 
122  // update the start sequence number
123  shift(length);
124  length = 0;
125  }
126  }
127  return request_len - length;

169 bool TCPRecvWindow::add_to_buffer_helper(uint32 seqno, uint32 length, byte* msg)
170 {
171  /* Copy the received message to the buffer, if msg is NULL, we will
172  take it as fake data. Here we assume that the packet either
173  overlaps completely with a packet already received, or is a
174  totally new packet. In other words, the sender will never send out
175  a partially overlapped packet. The method returns true if the data
176  is new. Otherwise, it returns false. */
177 
178  if(length <= 0) return false;
179  highest_seqno = mymax(highest_seqno, seqno + length - 1);
180 
181  // we use a linked list to put all segments.
182  if(!seg_buffer)
183  {
184  // there is no segment in the buffer
185  TCPSegment* seg = get_a_segment();
186  seg->set(seqno, length, msg, 0);
187  seg_buffer = seg;
188  return true; // a new packet
189  }
190  else if(seqno < seg_buffer->seqno)
191  {
192  // if the newly arrived is a packet predates all received ones
193  if(seqno + length == seg_buffer->seqno && !msg && !seg_buffer->msg)
194  {
195  // the new packet is just before the head packet in the buffer
196  // and both of them use fake data. we can merge them together.
197  seg_buffer->seqno = seqno;
198  seg_buffer->length += length;
199  }
200  else
201  {
202  TCPSegment* seg = get_a_segment();
203  seg->set(seqno, length, msg, seg_buffer);
204  seg_buffer = seg;
205  }
206  return true; // a new packet
207  }
208  else if(seqno < seg_buffer->seqno + length)
209  {
210  // this must be an old packet; according to our assumption, there
211  // must be a complete match
212  return false;
213  }
214  else
215  {
216  // the new packet is after the already-received packet.
217  TCPSegment* seg;
218  for(seg = seg_buffer; seg->next && seqno >= seg->next->seqno; seg = seg->next);
219  // check whether there is overlapping. when the loop stops, it
220  // must be either seg->next is NULL or seqno < seg->next->segno;
221  // therefore, the new packet can only overlap with the segment
222  // pointed by seg.
223  if(seqno < seg->seqno + seg->length)
224  {
225  // overlapping with an old packet
226  return false;
227  }
228  else if(seqno == seg->seqno + seg->length && !msg && !seg->msg)
229  {
230  // contiguous packets and they both use fake data.
231  seg->length += length;
232  return true;
233  }
234  else
235  {
236  // non-overlapping, non-contiguous, just put it behind seg.
237  TCPSegment* new_seg = get_a_segment();
238  new_seg->set(seqno, length, msg, seg->next);
239  seg->next = new_seg;
240  return true;
241  }
242  }
243 }
