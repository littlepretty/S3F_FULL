111 {
112  if(length <= 0) return 0;
113 
114  uint32 bytes_sent = 0;
115  if(sndwnd->used() < mymin(rcvwnd_size, cwnd))
116  {
117  uint32 nbytes = mymin(rcvwnd_size, cwnd)-sndwnd->used();
118  nbytes = mymin(nbytes, sndwnd->unused());
119  nbytes = mymin(nbytes, length);
120  nbytes = mymin(nbytes, mss);
121  nbytes = mymin(nbytes, S3FNET_IP_MAX_DATA_SIZE - S3FNET_TCPHDR_LENGTH);
122  while(nbytes == mss ||
123  0 < nbytes && nbytes < mss &&

125  {

127 
128  send_data(seqno, nbytes, 0, rcvwnd->expect(), true, true);

132  bytes_sent += nbytes;

135  nbytes = mymin(nbytes, sndwnd->unused());

144 uint32 TCPSession::resend_segments(uint32 seqno, uint32 nsegments)
145 {
146  // first cancel retransmission timeout
147  rxmit_timer_count = 0;
148 
149  // send up to nsegments
150  uint32 bytes_sent = 0;
151  nsegments = mymin(nsegments, (cwnd - sndwnd->lengthTo(seqno))/mss);
152  while(nsegments > 0)
153  {
154  // figure out minimum sized segment; minimum between mss and the
155  // amount of data to send in window
156  uint32 nbytes = mymin(mss, recover_seq - seqno);
157  if(nbytes > 0)
158  {
159  send_data(seqno, nbytes, 0, rcvwnd->expect(), false, true);
160  seqno += nbytes;
161  bytes_sent += nbytes;
162  nsegments--;
163  } else break;
164  }
165 
166  // update rxmit_seq if nesessary
167  rxmit_seq = seqno;
168 
169  // return nubmer for bytes sent
170  return bytes_sent;

191 void TCPSession::timeout_resend()
192 {
193  switch(state) {
194  case TCP_STATE_SYN_SENT:
195  // retransmit the syn segment
196  send_data(sndwnd->firstUnused(), 0, TCPMessage::TCP_FLAG_SYN,
197  rcvwnd->expect(), false, true);
198  break;
199 
200  case TCP_STATE_SYN_RECEIVED:
201  // retransmit the syn and ack segment
202  send_data(sndwnd->firstUnused(), 0, TCPMessage::TCP_FLAG_SYN | TCPMessage::TCP_FLAG_ACK,
203  rcvwnd->expect(), false, true);
204  break;
205 
206  case TCP_STATE_ESTABLISHED:
207  case TCP_STATE_CLOSE_WAIT:
208  // retransmit data segment
209  switch(tcp_master->getVersion()) {
210  case TCPMaster::TCP_VERSION_TAHOE:
211  resend_segments(sndwnd->start(), INT_MAX);
212  break;
213  case TCPMaster::TCP_VERSION_RENO:
214  case TCPMaster::TCP_VERSION_NEW_RENO:
215  resend_segments(sndwnd->start(), 1);
216  break;
217  case TCPMaster::TCP_VERSION_SACK:
218  sack_resend_segments(true);
219  break;
220  default:
221  assert(0);
222  }
223  break;
224 
225  case TCP_STATE_CLOSING:
226  case TCP_STATE_FIN_WAIT_1:
227  case TCP_STATE_LAST_ACK:
228  // resend the FIN segment
229  send_data(sndwnd->firstUnused/*next*/(), 0, TCPMessage::TCP_FLAG_FIN,
230  rcvwnd->expect(), false, true);
231  break;
232 
233  default:
234  // TCP_STATE_LISTEN, TCP_STATE_CLOSED, TCP_STATE_TIME_WAIT,
235  // or TCP_STATE_FIN_WAIT_2
236  error_quit("PANIC: TCPSession::timeout_resend(), unexpected state: %d.\n", state);
237  }
238 }

240 uint32 TCPSession::calc_advertised_wnd()
241 {
242  // first check the free space in the buffer
243  uint32 window = rcvwnd->freeInBuffer();
244  if(state < TCP_STATE_ESTABLISHED) return window;
245 
246  // if silly-window-syndrome-avoidance algorithm is used, run it
247  if(window * 4 < rcvwnd->bufferSize() || window < mss)
248  window = 0;
249 
250  // never shrink an already advertised window
251  window = mymax(window, rcvwnd->size());
252 
253  // update the current receive window
254  int incremental = window - rcvwnd->size();
255  if(incremental > 0) rcvwnd->expand(incremental);
256 
257  return window;
258 }
