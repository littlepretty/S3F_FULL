95 /*
96  * Performs a fast longest-matching prefix search for the key. The
97  * reason to do so is that it's used in routing entry searches.
98  * Because it's so critical to routing performance, a fast search is
99  * necessary.
100  *
101  * We assume the search length is independent. that means we try to
102  * find the best.
103  */
104 TrieData* Trie1::lookup(TRIEBITSTRING key)
105 {
106  // If this Trie exists to call 'lookup', root has already been asserted as non-null
107  TrieNode* cur = root->children[(key & TRIE_BIT_MASK)?1:0];
108  TrieData* pData = root->data; // handles default-route case
109 
110  while(cur) {
111  if(cur->data) {
112  pData = cur->data;
113  }
114  key <<= 1; // needed from pre-loop lookup and removes last unnecessary shift
115  cur = cur->children[(key & TRIE_BIT_MASK)?1:0];
116  }
117 
118  return pData;
119 }

152 /*
153  * Helper function for Insert and Lookup. This function will search
154  * the trie for the key. If found, it will return a pointer to the
155  * TrieNode, set failedNode to its parent, and set failedbitpos to the
156  * index in the parent's array. If not found, it will return NULL, and
157  * set failedNode to the TrieNode where the search fails and
158  * failedbitpos to the bit position that failed.
159  */
160 TrieNode* Trie1::search(TRIEBITSTRING key, int len,
161  TrieNode** failedNode, int* failedbitpos)
162 {
163  // TODO: Optimize this function
164  TrieNode* cur, *lastcur, *parent;
165  TrieNode* match = 0;
166  int bitpos, curbit, parentbit;
167  if(len > TRIE_KEY_SIZE(key)) {
168  len = TRIE_KEY_SIZE(key);
169  }
170 
171  // Go through the bits starting from the MSB and traverse
172  // the tree. Remember the last match.
173  for(bitpos = 0, lastcur = cur = root;
174  cur && bitpos < len;
175  cur = cur->children[curbit], bitpos++) {
176  if(cur->data) {
177  parent = lastcur;
178  parentbit = GETBIT(bitpos-1,key);
179  match = cur;
180  }
181 
182  lastcur = cur;
183  curbit = GETBIT(bitpos, key);
184  }
185 
186  if(cur && cur->data) {
187  match = cur;
188  if(failedNode && failedbitpos) {
189  // we have a match on the last node, parent is lastcur
190  *failedNode = lastcur;
191  *failedbitpos = GETBIT(bitpos-1,key);
192  }
193  } else if(failedNode && failedbitpos) {
194  // if there is a match
195  if(match) {
196  *failedNode = parent;
197  *failedbitpos = parentbit;
198  } else { // no match
199  if(cur) {
200  *failedbitpos = bitpos;
201  *failedNode = cur;
202  } else {
203  *failedbitpos = bitpos - 1;
204  *failedNode = lastcur;
205  }
206  }
207  }
208 
209  return match;
210 }

347 void Trie1::iterator::nextNode() {
348  bool keepMoving = true;
349  while(keepMoving) {
350  NodeNext* nn = stateStack.top();
351  stateStack.pop();
352  currentNode = nn->node;
353  assert(currentNode);
354  switch (nn->next) {
355  case LEFT: {
356  stateStack.push(new NodeNext(nn->node, RIGHT)); // afterwards move right
357  TrieNode* leftChild = currentNode->children[LEFT];
358  if (leftChild) { // has left child?
359  stateStack.push(new NodeNext(leftChild, LEFT)); // descend left
360  TRIE1_DUMP(std::cout << currentNode << " push RIGHT, LEFT down, push LEFT (" <<
361  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
362  } else {
363  TRIE1_DUMP(std::cout << currentNode << " push RIGHT (" << stateStack.size() <<
364  ") - "; currentNode->dump(); std::cout << std::endl);
365  }
366  }
367  break;
368  case RIGHT: {
369  stateStack.push(new NodeNext(nn->node, UP)); // afterwards move up
370  TrieNode* rightChild = currentNode->children[RIGHT];
371  if (rightChild) { // has right child?
372  TRIE1_DUMP(std::cout << currentNode << " push UP, RIGHT down, push LEFT (" <<
373  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
374  stateStack.push(new NodeNext(rightChild, LEFT)); // afterwards move left
375  } else {
376  TRIE1_DUMP(std::cout << currentNode << " push UP (" << stateStack.size() <<
377  ") - "; currentNode->dump(); std::cout << std::endl);
378  }
379  }
380  break;
381  case UP:
382  // visit node
383  keepMoving = false;
384  TRIE1_DUMP(std::cout << currentNode << " visit (" << stateStack.size() <<
385  ") - "; currentNode->dump(); std::cout << std::endl);
386  assert(currentNode != myTrie->root || stateStack.size() == 0);
387  break;
388  default:
389  // ERROR: undefined state of trie iterator
390  throw State();
391  } // switch
392  delete nn;
393  } // while
394 }
