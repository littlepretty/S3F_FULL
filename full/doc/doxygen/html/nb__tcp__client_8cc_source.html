199 void NBTCPClientSession::non_blocking_loop(NBTCPClientSessionContinuation* const caller) {
200  caller->p_state = NBTCPClientSessionContinuation::NB_TCP_CLIENT_SESSION_MAIN_LOOP;
201  caller->status = NBTCPClientSessionContinuation::NB_TCP_CLIENT_STATUS_IDLE; // see comment below
202  int res = sm->connect(caller->client_socket, server_ip, server_port, (NBSocketContinuation*)caller);
203  if(res == NBSocketMaster::ESUCCESS) {
204  connected = true;
205  }
206 
207  if(connected && file_size > bytes_received) {
208  int to_recv = file_size-bytes_received;
209  res = sm->recv(caller->client_socket, to_recv, 0, (NBSocketContinuation*)caller);
210  if(res == NBSocketMaster::ESUCCESS) {
211  bytes_received += caller->retval;
212  TCP_DUMP(printf("[host=\"%s\"] received %d of %d bytes\n",
213  inHost()->nhi.toString(),bytes_received,file_size));
214  /* This is a convoluted (and arguably very bad) way to do receive looping. It
215  * DOES, however, mirror the blocking socket's looping method. Ideally, this
216  * should just be called by caller->success() within recv(). Unfortunately,
217  * we cannot then predicate our looping on a successful receive, because success()
218  * would be called before we could see the return status, and with nonblocking
219  * sockets, we cannot guarantee that recv() will be called in the correct state.
220  * For example, recv() could be called before connect(), and we DO NOT want to set
221  * our continuation status to RECEIVING in that case.
222  */
223  caller->status = NBTCPClientSessionContinuation::NB_TCP_CLIENT_STATUS_RECEIVING;
224  caller->success();
225  } else if (res != NBSocketMaster::EWOULDBLOCK) {
226  TCP_DUMP(printf("[host=\"%s\"] encountered error %d\n",
227  inHost()->nhi.toString(),res));
228  }
229  }
230 
231  if(bytes_received >= file_size) {
232  caller->status = NBTCPClientSessionContinuation::NB_TCP_CLIENT_STATUS_IDLE;
233  sm->close(caller->client_socket, (NBSocketContinuation*)caller);
234  }
