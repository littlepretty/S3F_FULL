70 TrieData* Trie3::insert(TRIEBITSTRING key, int nBits, TrieData* data, bool replace) {
71  if(nBits > TRIE_KEY_SIZE(key)) { // the given mask is too long
72  fprintf(stderr, "Invalid mask size %d\n",nBits);
73  nBits = TRIE_KEY_SIZE(key);
74  }
75 
76  TrieNode* cur = root; // Note: This Trie implementation guarantees that root is never null
77  TrieNode* lastcur;
78  TRIEBITSTRING origkey = key;
79  int i;
80  int index;
81  // Note: This Trie implementation guarantees that any non-null, non-root node has data.
82  for(i=0;cur;i++) {
83  if(cur->data) { // TODO: Should be able to optimize this check out (see Note above)
84  // if this node already contains the correct forwarding data
85  if(data->equiv(cur->data)) {
86  // want to keep the node that may be propagated down the farthest
87  if(cur->maxDepth < nBits) {
88  cur->maxDepth = nBits;
89  cur->key = origkey;
90  }
91  delete data; // remove duplicate data (doesn't matter which since identical)
92  return 0;
93  // if inserting to this exact point but data is different, we may have a real conflict
94  } else if(nBits == i) {
95  /*
96  // DEBUG BLOCK
97  RouteInfo* currt = (RouteInfo*)cur->data;
98  RouteInfo* newrt = (RouteInfo*)data;
99  TRIE3_DUMP(printf("Potential conflict.\n");)
100  TRIE3_DUMP(printf("MATCH(nBits == %d): currt.data %s newrt.data\n",i,currt==newrt ? "==":"!=");)
101  TRIE3_DUMP(printf("inserted node: %s/%d, existing node: %s/%d\n",
102  IPPrefix::ip2txt(newrt->destination.addr), newrt->destination.len,
103  IPPrefix::ip2txt(currt->destination.addr), currt->destination.len);)
104  TRIE3_DUMP(printf("cur->maxDepth = %d, currt->dest.len = %d\n",cur->maxDepth,currt->destination.len);)
105  // END DEBUG BLOCK
106  */
107  // if the data was stored here solely as a space-optimization
108  if(cur->maxDepth > nBits) {
109  // try to propagate it down the tree
110  bool success = propagate(cur,nBits);
111  //TRIE3_DUMP(printf("propagate %s\n",success?"succeeded":"failed");) // DEBUG
112  if(success) {
113  // collision was avoided, insert given data at cur
114  cur->data = data;
115  cur->maxDepth = nBits;
116  cur->key = origkey;
117  return 0;
118  }
119  }
120 
121  if(!replace) {
122  return data;
123  }
124 
125  TrieData* temp = cur->data;
126  cur->data = data;
127  cur->maxDepth = nBits;
128  cur->key = origkey;
129  return temp;
130  }
131  }
132 
133  // check next node
134  key <<= 1;
135  lastcur = cur;
136  index = (key & TRIE_BIT_MASK)?1:0;
137  cur = cur->children[index];
138  }
139 
140  /*
141  // DEBUG BLOCK
142  RouteInfo* newrt = (RouteInfo*)data;
143  TRIE3_DUMP(printf("INSERTED node (depth == %d): %s/%d\n", i-1, IPPrefix::ip2txt(newrt->destination.addr), newrt->destination.len);)
144  // END DEBUG BLOCK
145  */
146 
147  // cur is null, so lastcur is where we need to insert
148  TrieNode* ptr = lastcur->children[index] = new TrieNode(origkey,nBits);
149  ptr->data = data;
150  nElements++;
151  return 0;
152 }

154 /*
155  * Performs a fast longest-matching prefix search for the key. The
156  * reason to do so is that it's used in routing entry searches.
157  * Because it's so critical to routing performance, a fast search is
158  * necessary.
159  *
160  * We assume the search length is independent. that means we try to
161  * find the best.
162  */
163 TrieData* Trie3::lookup(TRIEBITSTRING key)
164 {
165  // If this Trie exists to call 'lookup', root has already been asserted as non-null
166  TrieNode* cur = root->children[(key & TRIE_BIT_MASK)?1:0];
167  TrieData* pData = root->data; // handles default-route case
168 
169  while(cur) {
170  if(cur->data) {
171  pData = cur->data;
172  }
173  key <<= 1; // needed from pre-loop lookup and removes last unnecessary shift
174  cur = cur->children[(key & TRIE_BIT_MASK)?1:0];
175  }
176 
177  return pData;
178 }

180 /*
181  * Removes the key from the trie, if it exists, and returns the data
182  * corresponding to the removed key. If the key is not in the trie,
183  * return 0.
184  * In this Trie implementation, remove should be used with reservation as for
185  * nBits < TRIE_KEY_SIZE(key), multiple routes may actually be removed.
186  */
187 TrieData* Trie3::remove(TRIEBITSTRING key, int nBits) {
188  if(nBits > TRIE_KEY_SIZE(key)) { // the given mask is too long
189  fprintf(stderr, "Invalid mask size %d\n",nBits);
190  nBits = TRIE_KEY_SIZE(key);
191  }
192 
193  TrieNode* cur = root;
194  TrieNode* lastcur;
195  int i;
196  int index;
197  // move to target depth
198  for(i=0;cur && i < nBits;i++) {
199  key <<= 1;
200  lastcur = cur;
201  index = (key & TRIE_BIT_MASK)?1:0;
202  cur = cur->children[index];
203  }
204 
205  // if there is a match at the proper depth
206  if(cur && cur->data) {
207  TrieData* olddata = cur->data;
208  cur->data = 0;
209  // free this node if it is now a leaf
210  if(cur->children[0] == NULL && cur->children[1] == NULL) {
211  delete lastcur->children[index];
212  }
213 
214  return olddata;
215  }
216 
217  // no match was found at the given depth
218  return 0;
219 }

226 /* Recursively attempts to move a duplicate node with unique data down the Trie.
227  * TODO: May be possible to eliminate some unnecessary propagation failures. */
228 bool Trie3::propagate(TrieNode* node, int start_depth) {
229  //printf("called propagate( %s/%d , %d )\n",IPPrefix::ip2txt(node->key),node->maxDepth,start_depth); // TRACE
230  if(start_depth == TRIE_KEY_SIZE(key)) {
231  return false;
232  }
233 
234  TrieNode* cur = node;
235  TrieNode* temp;
236  int depth, curbit;
237  for(depth=start_depth+1; depth <= node->maxDepth; depth++) {
238  curbit = GETBIT(node->key,start_depth+1);
239  temp = cur->children[curbit];
240  if(temp == NULL) {
241  //TRIE3_DUMP(printf("found null node @depth %d\n",depth);) // DEBUG
242  temp = cur->children[curbit] = new TrieNode(node->key,node->maxDepth);
243  temp->data = node->data;
244  return true;
245  }
246  cur = temp;
247  }
248 
249  // propagated as deep as possible - must try to chain-propagate the node at maxDepth
250  if(cur->maxDepth > node->maxDepth) {
251  bool success = propagate(cur,depth);
252  if(success) {
253  cur->data = node->data;
254  cur->maxDepth = node->maxDepth;
255  cur->key = node->key;
256  }
257 
258  return success;
259  }
260 
261  return false;
262 }

381 void Trie3::iterator::nextNode() {
382  bool keepMoving = true;
383  while(keepMoving) {
384  NodeNext* nn = stateStack.top();
385  stateStack.pop();
386  currentNode = nn->node;
387  assert(currentNode);
388  switch (nn->next) {
389  case LEFT: {
390  stateStack.push(new NodeNext(nn->node, RIGHT)); // afterwards move right
391  TrieNode* leftChild = currentNode->children[LEFT];
392  if (leftChild) { // has left child?
393  stateStack.push(new NodeNext(leftChild, LEFT)); // descend left
394  TRIE3_DUMP(std::cout << currentNode << " push RIGHT, LEFT down, push LEFT (" <<
395  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
396  } else {
397  TRIE3_DUMP(std::cout << currentNode << " push RIGHT (" << stateStack.size() <<
398  ") - "; currentNode->dump(); std::cout << std::endl);
399  }
400  }
401  break;
402  case RIGHT: {
403  stateStack.push(new NodeNext(nn->node, UP)); // afterwards move up
404  TrieNode* rightChild = currentNode->children[RIGHT];
405  if (rightChild) { // has right child?
406  TRIE3_DUMP(std::cout << currentNode << " push UP, RIGHT down, push LEFT (" <<
407  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
408  stateStack.push(new NodeNext(rightChild, LEFT)); // afterwards move left
409  } else {
410  TRIE3_DUMP(std::cout << currentNode << " push UP (" << stateStack.size() <<
411  ") - "; currentNode->dump(); std::cout << std::endl);
412  }
413  }
414  break;
415  case UP:
416  // visit node
417  keepMoving = false;
418  TRIE3_DUMP(std::cout << currentNode << " visit (" << stateStack.size() <<
419  ") - "; currentNode->dump(); std::cout << std::endl);
420  assert(currentNode != myTrie->root || stateStack.size() == 0);
421  break;
422  default:
423  // ERROR: undefined state of trie iterator
424  throw State();
425  } // switch
426  delete nn;
427  } // while
428 }
