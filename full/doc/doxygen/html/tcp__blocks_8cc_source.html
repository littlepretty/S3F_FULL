40 bool TCPBlockList::overlap_blocks()
41 {
42  bool no_overlap = true;
43  TCPBlock *real_head = head_block->next, *p = real_head;
44 
45  // not overlapped if the list is empty
46  if(!p) return false;
47 
48  while(p->next)
49  {
50  // example: [100,200]->[0,100]
51  if((real_head->left_edge >= p->next->left_edge) &&
52  (real_head->left_edge <= p->next->right_edge))
53  {
54  no_overlap = false;
55 
56  // combine of the two blocks
57  real_head->left_edge = p->next->left_edge;
58  real_head->right_edge = mymax(real_head->right_edge, p->next->right_edge);
59  }
60 
61  // example: [0,100]->[100,200]
62  else if((real_head->right_edge >= p->next->left_edge) &&
63  (real_head->right_edge <= p->next->right_edge))
64  {
65  no_overlap = false;
66 
67  real_head->right_edge = p->next->right_edge;
68  real_head->left_edge = mymin(real_head ->left_edge, p->next->left_edge);
69  }
70

72  else

75  // block of the list

87 {
