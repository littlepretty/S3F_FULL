354 {
355  if(!str) return false;
356 
357  char* p= (char*)str;
358  char* end = p;
359 
360  if(*end == '\0') return false;
361 
362  CidrBlock* cur_cidr = &top_cidr_block;
363  for(;;)
364  {
365  long cur_id = strtol(p, &end, 10);

367  if(!sub_cidr) return false;

369  // move down.

371 
372  // check end condition
373  if(*end == '\0')

376  return true;
