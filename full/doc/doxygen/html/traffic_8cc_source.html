79 void Traffic::resolve_nhi()
80 {
81  // Now we will resolve each of the nhi's in the servers vector of
82  // each pattern. This must be done after IP addresses have been
83  // assigned to network interfaces.
84 
85  if(!patterns) return;
86 
87  for(TRAFFIC_PATTERN_LIST_ITERATOR it = patterns->begin(); it != patterns->end(); it++)
88  {
89  TrafficPattern* pattern = *it;
90 
91  // first, try to resolve the client's nhi address; we first try to
92  // interpret the client address as a network nhi, if no, we try
93  // machine nhi.
94  bool is_success = false;
95 
96  int nhi_type = pattern->client.type;
97  if(pattern->client.type & Nhi::NHI_NET)
98  {
99  pattern->client.type = Nhi::NHI_NET;
100  if(topnet->nhi2obj(&pattern->client))
101  is_success = true;
102  }
103 
104  if(!is_success)
105  {
106  pattern->client.type = nhi_type;
107  if(pattern->client.type & Nhi::NHI_MACHINE)
108  {
109  pattern->client.type = Nhi::NHI_MACHINE;
110  if(topnet->nhi2obj(&pattern->client))
111  is_success = true;
112  }
113  }

200 void TrafficPattern::configure_servers(s3f::dml::Configuration* cfg)
