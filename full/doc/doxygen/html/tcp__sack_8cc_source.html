40 int TCPSession::sender_send_sack(int sack_flag, byte mask, byte*& options)
41 {
42  // sender notifies the other side that sack is permitted.
43  if(sack_flag == TCP_SACK_PERMITTED)
44  {
45  // TCP_SACK_PERMITTED when SYN is set,otherwise terminate the program.
46  if(!(TCPMessage::TCP_FLAG_SYN & mask))
47  error_quit("ERROR: TCPSession::sender_send_sack(), "
48  "TCP_SACK_PERMITTED must be used together with SYN mask.\n");
49 
50  // because the header should be a multiple of 32 bits, so 4 bytes are here.
51  byte* sack_permitted_option = new byte[4];
52 
53  sack_permitted_option[0] = TCP_SACK_PERMITTED_KIND; // Kind = 4
54  sack_permitted_option[1] = 2; // Length = 2
55  options = sack_permitted_option;
56 
57  return 4; // add 4 bytes.
58  } else return 0;

61 int TCPSession::receiver_send_sack(int sack_flag, byte mask, uint32 ackno, byte*& options)
62 {
63  // as RFC 2018, SACK options should be included in all ACKs which do not ACK
64  // the highest seqnuence number in the data receiver's queue.
65  if((TCP_SACK_OPTION == sack_flag) &&(ackno < rcvwnd->getHighestSeqno()))
66  {
67  // the first 2 bytes are empty. 2 -> Kind(5), length; 4 * (left, right)
68  // important: make it's multiple of 32 bits.
69  // byte tcp_option[4+8*4];
70  byte* tcp_option = new byte[4 + 8 * TCP_SACK_MAX_BLOCKS];
71 
72  // kind = 5
73  tcp_option[0] = TCP_SACK_OPTION_KIND;
74  uint32 *buffer = (uint32 *)&tcp_option[2];
75 
76  // before sending out attached sack blocks, remove lowest
77  rcv_scoreboard->clear_blocks(ackno);
78 
79  // we can convey TCP_SACK_MAX_BLOCKS blocks
80  int real_num = rcv_scoreboard->fetch_blocks(buffer, TCP_SACK_MAX_BLOCKS);
81 
82  // update the length of valid sack option. totally, there are real_num blocks.
83  tcp_option[1] = 2 + 8 * real_num;
84 
85  // return added length
86  // attention: 4 + 8 * real_num includes the 2 padding bytes, which make
87  // the length be a multiple of 4 bytes.
88  // data->insert_at_front(tcp_option, 4 + 8 * real_num);
89  // return 1 + 2 * real_num; // (1 + 2 * real_num) * (32 bits)
90  //
91  // [03/23/04] always carry the same size.
92  options = tcp_option;
93  return 4 + 8 * TCP_SACK_MAX_BLOCKS;
94  }
95 
96  return 0;
97 }

99 void TCPSession::sender_recv_sack(TCPMessage* tmsg)
100 {
101  // update the sack information obtained.
102  if(tmsg->options && TCP_SACK_OPTION_KIND == tmsg->options[0])
103  {
104  TCP_DUMP(printf("sender_recv_sack(), SACK option set!\n"));
105  assert(tcp_master->getVersion() == TCPMaster::TCP_VERSION_SACK);
106 
107  // first if ack advances, we update the mySack_Entries.
108  if(tmsg->flags & TCPMessage::TCP_FLAG_ACK)
109  {
110  snd_scoreboard->clear_blocks(tmsg->ackno);
111  }
112 
113  // sack_length is 2 + 8 * num
114  // int num = ((tcp_hdr_option_length(msg)-2)/8);
115  assert(tmsg->options);
116  int num = (tmsg->options[1] - 2) / 8;
117  uint32 left,right;
118 
119  // now we sack the entries
120  for(int i=0; i<num; i++)
121  {
122  // 2 is (kind, length)
123  left = *((uint32 *)&tmsg->options[2 + 4 * 2 * i]);
124  right = *((uint32 *)&tmsg->options[2 + 4 * (2 * i + 1)]);
125  snd_scoreboard->insert_block(left, right - left);
126  }
127  }

162 void TCPSession::sack_send_in_fast_recovery()
