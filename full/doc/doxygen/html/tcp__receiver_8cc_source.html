147 {
148  switch(state) {

150  rcvwnd->setStart(tcphdr->seqno);

184 {
185  switch(state) {
186  case TCP_STATE_ESTABLISHED:
187  rcvwnd->setFIN(true);
188  // the other side wants to end the connection, we want to finish

192  break;
193

203  break;

359 void TCPSession::process_new_acks(TCPMessage* tcphdr, uint32 acked)
360 {
361  if(state == TCP_STATE_ESTABLISHED ||
362  state == TCP_STATE_CLOSE_WAIT)
363  {
364  // shift the window
365  sndwnd->release(acked);
366  }
367 
368  // As implemented in Java SSFNet, when new data got acked, first
369  // update retransmission timeout, and then update rtt. Is it right?
370 
371  // we don't need the retransmission timer any more
372  rxmit_timer_count = 0;
373 
374  // if there's still packet on the fly, we should reset the timeout
375  if(sndwnd->used() > 0)
376  {
377  // reschedule timer
378  rxmit_timer_count = timeout_value();
379  }
380 
381  // update timers. if data acked was RXMIT data then do not update
382  // (Karn's algorithm) Smoothed RTT and Dev calculations
383  if(rtt_count && (tcphdr->ackno > measured_seq))
384  {
385  // Jakobson's Algorithm for timers (1988)
386  rtt_measured = rtt_count - 1;
387  update_timeout();
388  }
389 
390  if(state != TCP_STATE_ESTABLISHED &&
391  state != TCP_STATE_CLOSE_WAIT)
392  return;
393 
394  // indicate whether it's the first partial ack
395  bool is_FirstNonDupACK;
396  if(ndupacks >= TCP_DEFAULT_MAX_DUPACKS)
397  {
398  // when the first non-duplicate ACK has arrived, enter fast
399  // recovery
400  is_FirstNonDupACK = true;
401 
402  switch(tcp_master->getVersion()) {
403  case TCPMaster::TCP_VERSION_TAHOE:
404  // do nothing, just go to slow start!
405  break;
406 
407  case TCPMaster::TCP_VERSION_RENO:
408  // reset congestion window and go to congestion avoidance
409  cwnd = mymin(ssthresh, tcp_master->getMaxConWnd());
410  // we check whether there are new data to send out we don't need
411  // to resend any segments here
412  break;
413 
414  case TCPMaster::TCP_VERSION_NEW_RENO:
415  // still stay in fast recovery, until acks of new data received
416  // we use fast_recovery to indicate it's still in fast
417  // recovery
418  fast_recovery = true;
419  break;
420 
421  case TCPMaster::TCP_VERSION_SACK:
422  fast_recovery = true;
423  break;
424 
425  default:
426  assert(0);
427  }
428  }
429  else is_FirstNonDupACK = false;
430 
431  // if no new data has been confirmed, we have to resend segments
432  if(tcphdr->ackno < recover_seq)
433  {
434  switch(tcp_master->getVersion()) {
435  case TCPMaster::TCP_VERSION_TAHOE:
436  // do nothing, just go to slow start!
437 
438  // The C++ TCP implementation is modified here to make it
439  // compatible with the Java version (and the NS-2 version as
440  // well). The congestion window is bumped by one each time a
441  // non-duplicate packet is received (we're in the slow start
442  // mood now). This happens no matter whether the ack is the
443  // first non-duplicate ack or not! -jason.
444  if(true || !is_FirstNonDupACK)
445  {
446  check_cwnd();
447  }
448 
449  // we resend the segments - GO-BACK-N
450  if(tcphdr->ackno > rxmit_seq)
451  {
452  resend_segments(tcphdr->ackno, UINT_MAX);
453  }
454  else
455  {
456  resend_segments(rxmit_seq, UINT_MAX);
457  }
458  break;
459 
460  case TCPMaster::TCP_VERSION_RENO:
461  // go to congestion avoidance phase
462  check_cwnd();
463 
464  if(timeout_loss)
465  {
466  if(tcphdr->ackno > rxmit_seq)
467  {
468  resend_segments(tcphdr->ackno, UINT_MAX);
469  }
470  else
471  {
472  resend_segments(rxmit_seq, UINT_MAX);
473  }
474  }
475  break;
476 
477  case TCPMaster::TCP_VERSION_NEW_RENO:
478  // it's still a partial ack, we resend one segment indicated by
479  // the ack number
480  resend_segments(tcphdr->ackno, 1);
481  // update congestion window: deflate the congestion window by
482  // the amount of data acknowledged and then add one back and
483  // send one new segment if permitted by the new value of cwnd
484  cwnd -= (acked - mss);
485  break;
486 
487  case TCPMaster::TCP_VERSION_SACK:
488  sack_pipe -= mss;
489  sack_send_in_fast_recovery();
490  break;
491 
492  default:
493  assert(0);
494  }
495  }
496 
497  // we have to update congestion window
498  else
499  {
500  // no timeout loss
501  timeout_loss = false;
502  if(fast_recovery &&
503  (tcp_master->getVersion() == TCPMaster::TCP_VERSION_NEW_RENO ||
504  tcp_master->getVersion() == TCPMaster::TCP_VERSION_SACK))
505  {
506  cwnd = mymin(ssthresh, mss + sndwnd->used());
507  fast_recovery = false;
508 
509  // clear all blocks
510  if(tcp_master->getVersion() == TCPMaster::TCP_VERSION_SACK)
511  snd_scoreboard->clear_all_blocks();
512  }
513  else check_cwnd();
514  }

517 void TCPSession::process_dup_acks()
518 {
519  // another duplicate ack
520  ndupacks++;
521  if(ndupacks == TCP_DEFAULT_MAX_DUPACKS)
522  {
523  calc_threshold();
524 
525  // we remember the largest seqno already sent out
526  recover_seq = sndwnd->firstUnused();
527  // we think this packet is lost, so we disable measuring rtt
528  rtt_count = 0;
529 
530  switch(tcp_master->getVersion()){
531  case TCPMaster::TCP_VERSION_TAHOE:
532  // in tahoe, fast recovery is off; tahoe fast rxmit resets
533  // congestion window to mss and resend 1 segment
534  cwnd = mss;
535  resend_segments(sndwnd->start(), 1);
536  break;
537 
538  case TCPMaster::TCP_VERSION_RENO:
539  case TCPMaster::TCP_VERSION_NEW_RENO:
540  // in Reno, fast recovery is on
541  cwnd = mymin(ssthresh + 3*mss, tcp_master->getMaxConWnd());
542  resend_segments(sndwnd->start(), 1);
543 
544  // more data to send out
545  if(sndwnd->unused() >= mss)
546  {
547  segment_and_send(sndwnd->firstUnused(),
548  mymin(sndwnd->unused(), sndwnd->dataInBuffer() - sndwnd->used()) );
549  }
550  break;
551 
552  case TCPMaster::TCP_VERSION_SACK:
553  sack_pipe = cwnd - TCP_DEFAULT_MAX_DUPACKS;
554  cwnd = mymin(ssthresh /* + 3*mss */, tcp_master->getMaxConWnd());
555  // we resend a packet
556  assert(sack_resend_segments(true));
557  break;
558 
559  default:
560  assert(0);
561  }
562  }
563 
564  else if(ndupacks > TCP_DEFAULT_MAX_DUPACKS)
565  {
566  // now we have more than TCP_DEFAULT_MAX_DUPACKS duplicate acks so if Reno
567  // then increase the Congestion Window by mss and resend another
568  // segment Do fast recovery
569  switch(tcp_master->getVersion()){
570  case TCPMaster::TCP_VERSION_TAHOE:
571  // do nothing
572  break;
573 
574  case TCPMaster::TCP_VERSION_RENO:
575  case TCPMaster::TCP_VERSION_NEW_RENO:
576  cwnd = mymin(cwnd + mss, tcp_master->getMaxConWnd());
577  if(sndwnd->unused() >= mss)
578  {
579  segment_and_send(sndwnd->firstUnused(),
580  mymin(sndwnd->unused(), sndwnd->dataInBuffer() - sndwnd->used()) );
581  }
582  break;
583 
584  case TCPMaster::TCP_VERSION_SACK:
585  sack_pipe -= mss;
586  sack_send_in_fast_recovery();
587  break;
588 
589  default:
590  assert(0);
591  }
592  }
