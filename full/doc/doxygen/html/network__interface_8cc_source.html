226  {
227  IFACE_DUMP(printf("[nhi=\"%s\", ip=\"%s\"] %s: receive a packet.\n",

229  }

231  //if the host is an openflow controller, handle the message (should be switch->controller message) directly

243  error_quit("NetworkInterface::receivePacket(), an OpenFlow switch received a packet on its controller interface, "
244  "but the packet type is not S3FNET_PROTOCOL_TYPE_OPENFLOW_CONTROLLER.");
245 
246  assert(of_switch_sess);
247  of_switch_sess->HandleControlInput(pkt);
248  return;
249  }
250

252  if(getHost()->isOpenVZEmu() == true)
253  {
254  emu_packet_modify_iface(pkt);
255  }
256 #endif
257 
258  // we pass the packet directly to the phy session
259  phy_sess->receivePacket(pkt, id);
