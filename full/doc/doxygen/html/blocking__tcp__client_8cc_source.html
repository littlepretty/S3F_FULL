67  /* When a continuation returns successfully, this function will be
68  called. Based on the current status of this continuation, we will
69  decide what to do next. */
70  virtual void success() {
71  BTCPClientSession* client = (BTCPClientSession*)owner;
72  switch(status) {
73  case BTCP_CLIENT_SESSION_CONNECTING:
74  client->server_connected(this);
75  break;
76  case BTCP_CLIENT_SESSION_RECEIVING:
77  client->data_received(this);
78  break;
79  case BTCP_CLIENT_SESSION_CLOSING:
80  client->session_closed(this);
81  delete this; // end of the journey!!!
82  break;
83  default:
84  TCP_DUMP(printf("ERROR: cnt=%lx received undefined success() status %d\n",(unsigned long)this,status)); // DEBUG
85  TCP_DUMP(printf("cnt->status @ %lx\n",(unsigned long)(&status))); // DEBUG
86  assert(0);
87  break;
88  }
89  }

91  /* When a continuation returns unsuccessfully, this function will be
92  called to print out the error. */
93  virtual void failure() {
94  BTCPClientSession* client = (BTCPClientSession*)owner;
95  switch(status) {
96  case BTCP_CLIENT_SESSION_CONNECTING:
97  TCP_DUMP(printf("BTCPClientSessionContinuation::failure() on host \"%s\": "
98  "failed to connect to server.\n",
99  client->inHost()->nhi.toString()));
100  break;
101  case BTCP_CLIENT_SESSION_RECEIVING: {
102  TCP_DUMP(printf("BTCPClientSessionContinuation::failure() on host \"%s\": "
103  "failed to receive data.\n",
104  client->inHost()->nhi.toString()));
105 
106  //if(user_timer && user_timer->isRunning())
107  if(user_timer)
108  {
109  // If the timer is running, it means that the failed receive
110  // is not due to time out so that the socket is aborted. In
111  // this case, we cancel the timer and manually abort the socket.
112  HandlePtr hptr(new Handle(user_timer)); //todo not sure isRunning is needed or not
113  hptr->cancel();
114  user_timer = 0;
115 
116  client->sm->abort(socket);
117  }
118 
119  client->main_proc(true, 0);
120  break;
121  }
122  case BTCP_CLIENT_SESSION_CLOSING:
123  TCP_DUMP(printf("BTCPClientSessionContinuation::failure() on host \"%s\": "
124  "failed to close connection.\n",
125  client->inHost()->nhi.toString()));
126  break;
127  default:
128  TCP_DUMP(printf("ERROR: received undefined failure() status %d\n",status));
129  TCP_DUMP(printf("cnt->status @ %lx\n",(unsigned long)(&status))); // DEBUG
130  assert(0);
131  break;
132  }

313 void BTCPClientSession::main_proc(int sample_off_time, ltime_t lead_time)
314 {
315  // check whether off time starts first
316  ltime_t t = lead_time;
317  if(sample_off_time)
318  {
319  if(off_time == 0)
320  {
321  TCP_DUMP(printf("[host=\"%s\"] %s: main_proc(), OFF (forever).\n",
322  inHost()->nhi.toString(), getNowWithThousandSeparator()));
323  return;
324  }
325  else
326  {
327  double vt_double;
328  if(off_time_exponential)
329  vt_double = getRandom()->Exponential(1.0/off_time_double);
330  else
331  vt_double = off_time_double;
332 
333  ltime_t vt = inHost()->d2t(vt_double, 0);
334  t += vt;
335 
336  TCP_DUMP(printf("[host=\"%s\"] %s: main_proc(), OFF (duration=%ld).\n",
337  inHost()->nhi.toString(), getNowWithThousandSeparator(), vt));
338  }
339  }
340 
341  Host* owner_host = inHost();
342  start_timer_callback_proc = new Process( (Entity *)owner_host, (void (s3f::Entity::*)(s3f::Activation))&BTCPClientSession::start_timer_callback);
343  start_timer_ac = new ProtocolCallbackActivation(this);
344  Activation ac (start_timer_ac);
345  HandleCode h = owner_host->waitFor( start_timer_callback_proc, ac, t, owner_host->tie_breaking_seed );

473 bool BTCPClientSession::get_random_server(IPADDR& server_ip, uint16& server_port)
474 {
475  Traffic* traffic = 0;

477 
478  //debug
479  /*printf("BTCPClientSession::get_random_server(), now = %s, before inNet()->control, ctrl_msg = %d\n",

481 
482  host->inNet()->control(NET_CTRL_GET_TRAFFIC, (void*)&traffic);
483  if(!traffic) return false;
484 
485  S3FNET_VECTOR(TrafficServerData*) servers;
486  if(!traffic->getServers(host, servers, server_list.c_str()) || servers.size() == 0)
487  {
488  if(show_report)
489  printf("WARNING: [host=\"%s\"] %s: get_random_server(), found no server for %s.\n",
490  inHost()->nhi.toString(), getNowWithThousandSeparator(), server_list.c_str());
491  return false;
492  }
493 
494  int sidx = (int)floor(getRandom()->Uniform(0, 1) * servers.size());
495  server_ip = host->inNet()->getNameService()->nhi2ip(servers[sidx]->nhi->toStlString());
496  server_port = servers[sidx]->port;
497 
498  if(server_ip == IPADDR_INVALID)
499  {
500  if(show_report)
501  printf("WARNING: [host=\"%s\"] %s: get_random_server(), unresolved server for \"%s\": %s:%d.\n",
502  inHost()->nhi.toString(), getNowWithThousandSeparator(), server_list.c_str(),
503  IPPrefix::ip2txt(server_ip), server_port);
504  return false;
505  }
506 
507  TCP_DUMP(printf("[host=\"%s\"] %s: get_random_server(), selected server: \"%s:%d\".\n",
508  inHost()->nhi.toString(), getNowWithThousandSeparator(), IPPrefix::ip2txt(server_ip), server_port));
509  return true;
