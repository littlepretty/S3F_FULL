166 int TCPSession::init_state_fin_wait_1()
167 {
168  state = TCP_STATE_FIN_WAIT_1;
169 
170  // make sure nothing to send out
171  assert(sndwnd->empty() && sndwnd->dataInBuffer() == 0);
172 
173  // Reset this since there is no data in window.
174  // In fact if its not 0 there's probably a bug.
175  nrxmits = 0;
176 
177  // Send a fin and possibly ack any left over delayed acks
178  if(tcp_master->isDelayedAck() && delayed_ack)
179  {
180  send_data(sndwnd->firstUnused(), 0,
181  TCPMessage::TCP_FLAG_FIN | TCPMessage::TCP_FLAG_ACK,
182  rcvwnd->expect(), true, true);
183  cancel_delay_ack();
184  tcp_master->disableFastTimeout(this);
185  }
186  else
187  {
188  send_data(sndwnd->firstUnused(), 0, TCPMessage::TCP_FLAG_FIN,
189  rcvwnd->expect(), true, true);
190  }
191 
192  sndwnd->setFIN(true);
193 
194  // Now if established state received a fin while was waiting to dump
195  // its buffer, then next state will be TCP_STATE_CLOSING.
196  if(simultaneous_closing) return init_state_closing();
197  else return 0;
