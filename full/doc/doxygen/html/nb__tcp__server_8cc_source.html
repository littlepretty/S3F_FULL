148 int FileSender::handleIO(NBSocketMaster* sm, NBTCPServerSessionContinuation* cnt) {
149  TCP_DUMP(printf("called handleIO() (socket=%d,mState=%d)\n",mSocket,mState));
150  if(mState == IDLE)
151  return 1; /* nothing to do */
152 
153  /* If buffer empty, fill it */
154  if(mBufUsed == mBufLen) {
155  /* Get one (fake) chunk of the file*/
156  mBufLen = mFileSize-mBytesSent;
157 /* We need to send the entire filesize at once with send(). Otherwise, we don't wake up properly.
158  * This behavior is consistent with the blocking implementation too. Alternatively, we could add
159  * a periodic callback to wake up this application and send more data.
160  if(mBufLen > BUFSIZE) {
161  mBufLen = BUFSIZE;
162  }
163  */
164  if(mBufLen == 0) {
165  /* All done; close the socket. */
166  int res = sm->close(mSocket,cnt);
167  TCP_DUMP(printf("close (#1) socket %d returned %d\n",mSocket,res));
168  mState = IDLE;
169  return 3;
170  }
171  mBufUsed = 0;
172  }
173 
174  /* Send one chunk of the file */
175  assert(mBufLen > mBufUsed);
176  int res = sm->send(mSocket, mBufLen-mBufUsed, 0, (NBSocketContinuation*)cnt);
177  //nwrite = write(mSocket, mBuf + mBufUsed, mBufLen - mBufUsed);
178  if(res == NBSocketMaster::ESUCCESS) {
179  if(cnt->retval < 0) {
180  TCP_DUMP(printf("send failed (error %d, retval = %d)!\n",res,cnt->retval));
181  } else {
182  TCP_DUMP(printf("%d bytes of %d bytes sent\n",mBytesSent+cnt->retval,mFileSize));
183  mBufUsed += cnt->retval;
184  mBytesSent += cnt->retval;
185  }
186  } else if(res != NBSocketMaster::EWOULDBLOCK) {
187  TCP_DUMP(printf("send failed (error %d, retval = %d)!\n",res,cnt->retval));
188  } else {
189  return 2;
190  }
191 
192  /* Due to lack of callbacks, we need this - with callbacks, the earlier check suffices */
193  if(mBytesSent >= mFileSize) {
194  /* All done; close the socket. */
195  int res = sm->close(mSocket,cnt);
196  TCP_DUMP(printf("close (#2) socket %d returned %d\n",mSocket,res));
197  mState = IDLE;
198  return 3;
199  }
200 
201  return 0;
202 }

242 /* NOTE: Without callback, only makes one pass at a time - sends 1 block to each connected instead
243 of looping, but this mirrors the blocking behavior as well; TCP handles the wake-and-send behavior */
244 void NBTCPServerSession::non_blocking_loop(NBTCPServerSessionContinuation* const caller) {
245  caller->p_state = NBTCPServerSessionContinuation::NB_TCP_SERVER_SESSION_MAIN_LOOP;
246  caller->status = NBTCPServerSessionContinuation::NB_TCP_SERVER_STATUS_IDLE; // see last comment in function
247  /* If we don't have a full set of clients already,
248  * listen for a new connection.
249  */
250  if(nclients < MAXCLIENTS) {
251  int res = sm->accept(caller->server_socket, true, (NBSocketContinuation*)caller, 0);
252  if(res == NBSocketMaster::ESUCCESS) {
253  int fd = caller->retval;
254  if(fd != -1) {
255  /* Someone connected. Send them the file */
256  TCP_DUMP(printf("[host=\"%s\"] new connection.\n", inHost()->nhi.toString()));
257  mFilesenders[nclients].sendFile(DEFAULT_REQ_SIZE,fd);
258  nclients++;
259  }
260  }
261  }
262  // TODO: Need to release filesenders at some point for reuse - exhausts after MAXCLIENTS
263  /* Pump data out to all the connected clients */
264  bool idle = true;
265  for (int i=0; i<nclients; i++) {
266  if(!mFilesenders[i].handleIO(sm,caller)) {
267  idle = false;
268  }
269  }
270 
271  /* NOTE: Somewhat similar to nb_tcp_client, this could be done MUCH better. This looping
272  * behavior, while correct, is convoluted and not at all elegant. */
273  if(!idle) {
274  caller->status = NBTCPServerSessionContinuation::NB_TCP_SERVER_STATUS_SENDING;
275  caller->success();
276  }
277 }
