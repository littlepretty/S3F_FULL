144 int UDPSession::send(int length, byte* msg)
145 {
146  int offset = 0;
147  while(offset < length) // if there's anything left to be sent
148  {
149  int to_send = mymin(udp_master->max_datagram_size, length-offset);
150 
151  UDP_DUMP(printf("[host=\"%s\"] %s: send(length=%d), send a message of %d bytes, offset=%d.\n",
152  udp_master->inHost()->nhi.toString(),
153  udp_master->getNowWithThousandSeparator(), length, to_send, offset));
154 
155  DataMessage* dmsg;
156  if(msg) dmsg = new DataMessage(to_send, &msg[offset], true);
157  else dmsg = new DataMessage(to_send);
158  offset += to_send;
159 
160  UDPMessage* udphdr = new UDPMessage(src_port, dst_port);
161  udphdr->carryPayload(dmsg);
162  Activation ac(udphdr);
163  IPPushOption ops(src_ip, dst_ip, S3FNET_PROTOCOL_TYPE_UDP, DEFAULT_IP_TIMETOLIVE);
164  udp_master->pushdown(ac, 0, (void*)&ops, sizeof(IPPushOption));
165  }
166 
167  wake_app(SocketSignal::OK_TO_SEND);
168  return length;
169 }

171 int UDPSession::recv(int length, byte* msg)
172 {
173  assert(length > 0);

184  clear_app_state(SocketSignal::DATA_AVAILABLE);
185  }

217 int UDPSession::generate(int length, byte* msg)
218 {
219  int rcvd = 0;
220  // scan thru the received data blocks
221  while(!rcvbuf.empty())
222  {
223  DataMessage* dmsg = rcvbuf.front();
224 
225  // data received not enough to fill application receive buffer
226  if(dmsg->realByteCount()-rcvbuf_offset <= length)
227  {
228  int to_recv = dmsg->realByteCount()-rcvbuf_offset;
229  if(msg && dmsg->payload)
230  {
231  memcpy(&msg[rcvd], (byte*)dmsg->payload+rcvbuf_offset, to_recv);
232  }
233  rcvbuf_len -= to_recv;
234  rcvbuf_offset = 0;
235  rcvd += to_recv;
236  rcvbuf.pop_front();
237  delete dmsg;
238 
239  if(to_recv == length) break;
240  }
241 
242  // the data message contains more bytes than needed by application
243  else
244  {
245  int to_recv = length;
246  if(msg && dmsg->payload)
247  {
248  memcpy(&msg[rcvd], (byte*)dmsg->payload+rcvbuf_offset, to_recv);
249  }
250  rcvbuf_len -= to_recv;
251  rcvbuf_offset += to_recv;
252  rcvd += to_recv;
253  break;
254  }
255  }
256  return rcvd;
257 }
