366 /* double Incomplete_Gamma() */
367 /* ========================================================================= */
368 /* Evaluates the incomplete gamma function. */
369 /* NOTE: use a > 0.0 and x >= 0.0 */
370 /* */
371 /* The algorithm used to evaluate the incomplete gamma function is based on */
372 /* Algorithm AS 32, J. Applied Statistics, 1970, by G. P. Bhattacharjee. */
373 /* See also equations 6.5.29 and 6.5.31 in the Handbook of Mathematical */
374 /* Functions, Abramowitz and Stegum (editors). The absolute error is less */
375 /* than 1e-10 for all non-negative values of x. */
376 /* ========================================================================= */
377 double RNG::Incomplete_Gamma(double a, double x)
378 { double tiny = 1.0e-10; /* convergence parameter */
379  double t, sum, term, factor, f, g, c[2], p[3], q[3];
380  long n;
381 
382  if (x > 0.0)
383  factor = exp(-x + a * log(x) - Ln_Gamma(a));
384  else
385  factor = 0.0;
386  if (x < a + 1.0) { /* evaluate as an infinite series - */
387  t = a; /* A & S equation 6.5.29 */
388  term = 1.0 / a;
389  sum = term;
390  do { /* sum until 'term' is small */
391  t++;
392  term *= x / t;
393  sum += term;
394  } while (term >= tiny * sum);
395  return(factor * sum);
396  }
397 
398  else { /* evaluate as a continued fraction - */
399  p[0] = 0.0; /* A & S eqn 6.5.31 with the extended */
400  q[0] = 1.0; /* pattern 2-a, 2, 3-a, 3, 4-a, 4,... */
401  p[1] = 1.0; /* - see also A & S sec 3.10, eqn (3) */
402  q[1] = x;
403  f = p[1] / q[1];
404  n = 0;
405  do { /* recursively generate the continued */
406  g = f; /* fraction 'f' until two consecutive */
407  n++; /* values are small */
408  if ((n%2) > 0) {
409  c[0] = ((double)(n + 1) / 2) - a;
410  c[1] = 1.0;
411  }
412  else {
413  c[0] = (double)(n / 2);
414  c[1] = x;
415  }
416  p[2] = c[1] * p[1] + c[0] * p[0];
417  q[2] = c[1] * q[1] + c[0] * q[0];
418  if (q[2] != 0.0) { /* rescale to avoid overflow */
419  p[0] = p[1] / q[2];
420  q[0] = q[1] / q[2];
421  p[1] = p[2] / q[2];
422  q[1] = 1.0;
423  f = p[1];
424  }
425  } while ((fabs(f - g) >= tiny) || (q[1] != 1.0));
426  return((double)1.0 - factor * f);
427  }
428 } /* Incomplete_Gamma */

432 /* ========================================================================= */
433 /* Evaluates the incomplete beta function. */
434 /* NOTE: use a > 0.0, b > 0.0 and 0.0 <= x <= 1.0 */
435 /* */
436 /* The algorithm used to evaluate the incomplete beta function is based on */
437 /* equation 26.5.8 in the Handbook of Mathematical Functions, Abramowitz */
438 /* and Stegum (editors). The absolute error is less than 1e-10 for all x */
439 /* between 0 and 1. */
440 /* ========================================================================= */
441 double RNG::Incomplete_Beta(double a, double b, double x)
442 { double tiny = 1.0e-10; /* convergence parameter */
443  double t, factor, f, g, c, p[3], q[3];
444  int swap;
445  long n;
446 
447  if (x > (a + 1) / (a + b + 1)) { /* to accelerate convergence */
448  swap = 1; /* complement x and swap a & b */
449  x = 1 - x;
450  t = a;
451  a = b;
452  b = t;
453  }
454  else /* do nothing */
455  swap = 0;
456  if (x > 0)
457  factor = exp(a * log(x) + b * log(1 - x) - Ln_Beta(a,b)) / a;
458  else
459  factor = 0;
460  p[0] = 0;
461  q[0] = 1;
462  p[1] = 1;
463  q[1] = 1;
464  f = p[1] / q[1];
465  n = 0;
466  do { /* recursively generate the continued */
467  g = f; /* fraction 'f' until two consecutive */
468  n++; /* values are small */
469  if ((n%2) > 0) {
470  t = (n - 1) / 2;
471  c = -(a + t) * (a + b + t) * x / ((a + n - 1) * (a + n));
472  }
473  else {
474  t = n / 2;
475  c = t * (b - t) * x / ((a + n - 1) * (a + n));
476  }
477  p[2] = p[1] + c * p[0];
478  q[2] = q[1] + c * q[0];
479  if (q[2] != 0) { /* rescale to avoid overflow */
480  p[0] = p[1] / q[2];
481  q[0] = q[1] / q[2];
482  p[1] = p[2] / q[2];
483  q[1] = 1.0;
484  f = p[1];
485  }
486  } while ((ABS(f - g) >= tiny) || (q[1] != 1));
487  if (swap)
488  return(1.0 - factor * f);
489  else
490  return(factor * f);
491 } /* Incomplete_Beta */
