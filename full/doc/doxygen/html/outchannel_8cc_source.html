55 
56  This method write directs the passed event (Activation act)
57  to be carried to all inchannels to which
58  this outchannel is mapped. The written event is scheduled
59  to arrive at its destination at the time equal to the current time plus the offered
60  per-write delay plus the delay associated with the mapto connection.
61 
62  One form of write specifies a scheduling priority to be used as a 2nd key
63  in the event-list of the target timeline. If no priority is specified,
64  the default is the priority code of the process activated by this write.
65  Likewise, a per-write delay may or may not be included. If it is not, zero is assumed.
66 
67  A write attempts to write a copy of Activation act to every InChannel targeted by
68  a mapto call by this OutChannel. Any InChannel that is on this same
69  Timeline will get its Activation. Any write done where the
70  per-write delay is at least as large as the minimum declared at construction
71  will deliver the Activation to all the targets. However, we allow a per-write
72  delay that is actually smaller than the promised minimum, provided that the
73  overall delivery time still places the receipt in a future synchronization window.
74  The upper edge of the current window is accessible through Timeline::horizon().
75  A write that would try to place the Activation across Timelines and within the
76  current window is not performed, but all legal deliveries eminating from the write
77  will be delivered. True is returned if all deliveries are made, False if one or more
78  do not. A model may test the return from write and has access to enough information
79  to determine which deliveries were not made.
80 
81  Activation is typedef for shared_ptr<Message>. Message is the base class for
82  a communication, shared_ptr templates a very capable smart pointer, that, in particular,
83  alleviates the need for deleting Activations as they are past around. In practice
84  a modeler will create a derived class for Message and create an Activation for it.
85  The Message base class carries an unsigned int type that can be used to encode the
86  derived type of Message being carried, needed for casting at the destination.

387  ltime_t OutChannel::new_transfer_delay(InChannel* ic, ltime_t delay)
388 
389  new_transfer_delay changes the transfer delay between the OutChannel
390  and named InChannel, provided the mapping exists. If it does not exist,
391  -1 is returned. If it does exist, a time by which the change will be
392  implemented is returned.
393  ****************************************************************/
394 ltime_t OutChannel::new_transfer_delay(InChannel* ic, ltime_t delay) {
395 
396  // cache for efficiency
397  int end = __mapped_to.size();
398 
399  Timeline* tl = alignment();
400 
401  // find the named InChannel
402  for(int i=0; i< end; i++) {
403  if( __mapped_to[i].ic() == ic) {
404 
405  // Found it. If the new minimum is equal to the old one there
406  // is nothing to do, just return the current time.
407  //
408  if( delay == __mapped_to[i].transfer_delay() ) return tl->now();
409 
410  // if the channel endpoints are co-aligned, just
411  // make the change immediately and return the current time
412  //
413  if( !__mapped_to[i].xtimeline() ) {
414  __mapped_to[i].set_transfer_delay(delay);
415  return tl->now();
416  }
417 
418  // If the new delay is larger than the old we can implement it
419  // knowing it does not affect synchronization
420  //
421  if( __mapped_to[i].transfer_delay() < delay ) {
422  __mapped_to[i].set_transfer_delay(delay);
423  return tl->now();
424  }
425 
426  // The delay got smaller. compute the new minimum transfer time under this change
427  ltime_t xtransfer = min_write_delay() + delay;
428 
429  // if the new minimum is equal to the timeline's
430  // minimum channel delay then increment the number
431  // of channels at the minimum
432  //
433  if( xtransfer == tl->get_min_sync_cross_timeline_delay() ) {
434  tl->__channels_at_minimum++;
435  __mapped_to[i].set_transfer_delay(delay);
436  return tl->now();
437  }
438 
439  // if the new minimum is less than the timeline's
440  // minimum channel delay it may still be possible to implement,
441  // if
442  // -- the timeline isn't currently running
443  // -- we've advanced far enough into the synchronization window
444  // that we can implement the change without threatening
445  // the synchronization
446  //
447  if( xtransfer < tl->get_min_sync_cross_timeline_delay() &&
448  ((tl->get_state() != RUNNING) ||
449  ( tl->horizon() <= tl->now()+xtransfer ))) {
450  __mapped_to[i].set_transfer_delay(delay);
451  return tl->now();
452  }
453 
454  // necessary to delay implementation and recompute the minimum cross timeline delay
455  tl->__recompute_min_delay = true;
456 
457  // Create a mappedChannel structure and place it in the
458  // timeline's __delays_to_change vector
459  //
460  tl->__delays_to_change.push_back(
461  new mappedChannel( this, ic, delay, false) );
462  break;
463  }
464  }
465  return tl->horizon();
466 }
