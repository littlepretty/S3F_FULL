70 TrieData* Trie0::insert(TRIEBITSTRING key, int nBits,
71  TrieData* data, bool replace)
72 {
73  TrieNode *fail = 0;
74  int bitpos = 0;
75 
76  TrieNode* match = search(key, nBits, &fail, &bitpos);
77 
78  // Does the key already exist in our trie?
79  // If so, replace the data if replace flag is true.
80  if(match) {
81  if(!replace) return data;
82  TrieData* temp = match->data;
83  match->data = data;
84  return temp;
85  }
86 
87  // Create all the necessary nodes for the unmatched bits
88  // after fail and insert the data.
89  create(fail, REMOVE(bitpos, key), nBits-bitpos, data);
90  nElements++;
91 
92  return 0;
93 }

95 /*
96  * Performs a fast longest-matching prefix search for the key. The
97  * reason to do so is that it's used in routing entry searches.
98  * Because it's so critical to routing performance, a fast search is
99  * necessary.
100  *
101  * We assume the search length is independent. that means we try to
102  * find the best.
103  */
104 TrieData* Trie0::lookup(TRIEBITSTRING key) {
105  TrieNode* cur = root;
106  TrieData* pData = 0;
107  while(cur) {
108  if(cur->data) pData = cur->data;
109  cur = cur->children[(key&0x80000000)?1:0];
110  key <<= 1;
111  }
112 
113  return pData;
114 }

145 /*
146  * Helper function for Insert and Lookup. This function will search
147  * the trie for the key and if found, return a pointer to the
148  * TrieNode. If not found, it will return NULL, and set failedNode to
149  * the TrieNode where the search fails and failedbitpos to the bit
150  * position that failed.
151  */
152 TrieNode* Trie0::search(TRIEBITSTRING key, int len,
153  TrieNode** failedNode, int* failedbitpos)
154 {
155  TrieNode* cur, *lastcur;
156  TrieNode* match = 0;
157  int bitpos, curbit;
158 
159  // Go through the bits starting from the MSB and traverse
160  // the tree. Remember the last match.
161  for(bitpos = 0, lastcur = cur = root;
162  cur && bitpos < len && bitpos < TRIE_KEY_SIZE(key);
163  cur = cur->children[curbit], bitpos++) {
164  if(cur->data) match = cur;
165 
166  lastcur = cur;
167  curbit = GETBIT(bitpos, key);
168  }
169 
170  if(cur && cur->data) match = cur;
171  else if(failedNode && failedbitpos) {
172  if(cur) {
173  *failedbitpos = bitpos;
174  *failedNode = cur;
175  } else {
176  *failedbitpos = bitpos - 1;
177  *failedNode = lastcur;
178  }
179  }
180 
181  return match;
182 }

319 void Trie0::iterator::nextNode() {
320  bool keepMoving = true;
321  while(keepMoving) {
322  NodeNext* nn = stateStack.top();
323  stateStack.pop();
324  currentNode = nn->node;
325  assert(currentNode);
326  switch (nn->next) {
327  case LEFT: {
328  stateStack.push(new NodeNext(nn->node, RIGHT)); // afterwards move right
329  TrieNode* leftChild = currentNode->children[LEFT];
330  if (leftChild) { // has left child?
331  stateStack.push(new NodeNext(leftChild, LEFT)); // descend left
332  TRIE0_DUMP(std::cout << currentNode << " push RIGHT, LEFT down, push LEFT (" <<
333  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
334  } else {
335  TRIE0_DUMP(std::cout << currentNode << " push RIGHT (" << stateStack.size() <<
336  ") - "; currentNode->dump(); std::cout << std::endl);
337  }
338  }
339  break;
340  case RIGHT: {
341  stateStack.push(new NodeNext(nn->node, UP)); // afterwards move up
342  TrieNode* rightChild = currentNode->children[RIGHT];
343  if (rightChild) { // has right child?
344  TRIE0_DUMP(std::cout << currentNode << " push UP, RIGHT down, push LEFT (" <<
345  stateStack.size() << ") - "; currentNode->dump(); std::cout << std::endl);
346  stateStack.push(new NodeNext(rightChild, LEFT)); // afterwards move left
347  } else {
348  TRIE0_DUMP(std::cout << currentNode << " push UP (" << stateStack.size() <<
349  ") - "; currentNode->dump(); std::cout << std::endl);
350  }
351  }
352  break;
353  case UP:
354  // visit node
355  keepMoving = false;
356  TRIE0_DUMP(std::cout << currentNode << " visit (" << stateStack.size() <<
357  ") - "; currentNode->dump(); std::cout << std::endl);
358  assert(currentNode != myTrie->root || stateStack.size() == 0);
359  break;
360  default:
361  // ERROR: undefined state of trie iterator
362  throw State();
363  } // switch
364  delete nn;
365  } // while
366 }
