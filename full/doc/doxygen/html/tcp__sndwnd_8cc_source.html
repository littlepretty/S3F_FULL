38 DataMessage* TCPSendWindow::generate(uint32 seqno, uint32 length)
39 {
40  // make sure the sequence does not fall within the buffer
41  assert(seqno >= start() && seqno + length <= start() + length_in_buffer);
42 
43  // calculate the offset from the start
44  uint32 offset = seqno - start();
45 
46  // find the right block in the list
47  DataChunk* node = head_chunk;
48  while(node && offset)
49  {
50  // if the start position is at node->real_data + offset in the current
51  // node, quit the loop.
52  if(node->real_length > offset) break;
53  offset -= node->real_length;
54  node = node->next;
55  }
56  assert(node);
57 
58  // fetch data and put it into a generated payload message.
59  if(node->real_length - offset >= length)
60  {
61  byte* payload;
62  // the current node can satisfy the data length requested.
63  if(node->real_data)
64  {
65  // if the current node carries real data, it is the first node
66  // and, at the same time, the current node can satisfy the data
67  // length required
68  payload = new byte[length];
69  memcpy((byte*)payload, node->real_data+offset, length);
70  }
71  else
72  {
73  // if the current node carries fake data, it is the first node
74  // and, at the same time, the current node can satisfy the data
75  // length required
76  payload = 0;
77  }
78  return new DataMessage(length, payload);
79  }
80  else
81  {
82  // the current node cannot satisfy the data length requested; we
83  // maintain a message list, since we need to attach each new block
84  // at the tail of the list, we need to keep the tail pointer.
85  DataChunk* msglist_head = 0, *msglist_tail = 0;
86  uint32 cur_length = length;
87  while(cur_length > 0)
88  {
89  // for the current block, how much data can be transmitted?
90  int len = mymin(node->real_length-offset, cur_length);
91 
92  assert(node);
93  if(node->real_data)
94  {
95  // look ahead to see how many real data ahead; we want to consolidate the real data here...
96  int contiguous_len = len;
97  for(DataChunk* pnode = node->next; pnode && pnode->real_data; pnode = pnode->next)
98  {
99  if(pnode->real_length+contiguous_len >= cur_length)
100  {
101  contiguous_len += mymin(cur_length - contiguous_len, pnode->real_length);
102  break;
103  }
104  else contiguous_len += pnode->real_length;
105  }
106 
107  // we allocate a block containing these contiguous data
108  byte* block = new byte[contiguous_len];
109  int copied_len = 0;
110  while(copied_len < contiguous_len)
111  {
112  int new_len = mymin(contiguous_len-copied_len, (int)(node->real_length-offset));
113  memcpy(block+copied_len, node->real_data+offset, new_len);
114  copied_len += new_len;
115  if(offset+new_len < node->real_length)
116  {
117  offset += new_len;
118  assert(copied_len == contiguous_len);
119  }
120  else
121  {
122  node = node->next;
123  offset = 0;
124  }
125  }
126 
127  cur_length -= contiguous_len;
128  assert(offset==0 || (offset>0 && cur_length==0));
129 
130  // put the contiguous message block at the tail
131  DataChunk* msg_node = new DataChunk(contiguous_len, block);
132  if(!msglist_head)
133  {
134  msglist_head = msglist_tail = msg_node;
135  }
136  else
137  {
138  msglist_tail->next = msg_node;
139  msglist_tail = msg_node;
140  }
141  }
142  else // if fake data
143  {
144  if(!msglist_head)
145  {
146  msglist_head = msglist_tail = new DataChunk(len);
147  }
148  else if(!msglist_tail->real_data)
149  {
150  // the tail block also uses fake data, so we can merge them
151  // together by simply adding the lengths
152  msglist_tail->real_length += len;
153  }
154  else
155  {
156  // the tail block uses real data, we have to create another
157  // message node to contain this fake data
158  msglist_tail->next = new DataChunk(len);
159  msglist_tail = msglist_tail->next;
160  }
161 
162  // we have to visit the next message node if not enough data is fetched.
163  if((cur_length -= len) > 0)
164  {
165  node = node->next;
166  offset = 0;
167  }
168  }
169  } // while loop
170 
171  return new DataMessage(msglist_head);
172  }

242 void TCPSendWindow::release_buffer(uint32 length)
243 {
244  assert(length <= length_in_buffer);
245 
246  length_in_buffer -= length;
247  while(length > 0)
248  {
249  // check whether we can finish it in the current node.
250  if(head_chunk->real_length > length)
251  {
252  head_chunk->real_length -= length;
253  if(head_chunk->real_data)
254  head_chunk->real_data += length;
255  length = 0;
256  }
257  else // we need to destroy the current node
258  {
259  length -= head_chunk->real_length;
260 
261  // release the node; note that the real data (present or not)
262  // should not be reclaimed here and we should not chain the
263  // deletion!
264  DataChunk* node = head_chunk->next;
265  head_chunk->real_data = 0;
266  head_chunk->next = 0;
267  delete head_chunk;
268  head_chunk = node;
269 
270  if(first_raw_data) delete[] first_raw_data;
271  if(head_chunk)
272  {
273  first_raw_data = head_chunk->real_data;
274  }
275  else
276  {
277  // in case that the whole list is deleted
278  tail_chunk = 0;
279  first_raw_data = 0;
280  }
