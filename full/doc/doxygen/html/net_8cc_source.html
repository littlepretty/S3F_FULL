665 int Net::process_ARP_packet(EmuPacket *ppkt)
666 {
667  if(ppkt->len <= 0)
668  {
669  printf("Net::process_ARP_packet(), len = %d \n", ppkt->len);
670  return 0;
671  }
672 
673  char veip_target[32];
674  arp = (struct sniff_arp*)(ppkt->data + SIZE_ETHERNET);
675 
676  /* get the Target IP address */
677  sprintf (veip_target, "%u.%u.%u.%u", arp->__ar_tip[0],arp->__ar_tip[1],arp->__ar_tip[2],arp->__ar_tip[3]);
678  if(veip_target == NULL)
679  {

681  return 0;

684  IPADDR ip_target = IPPrefix::txt2ip(veip_target);
685 
686  if(ip_target == IPADDR_INVALID)
687  {
688  NET_DUMP(printf("Net::process_ARP_packet(): veip_target (%s) not found.\n", veip_target));
689  return 0;
690  }
691  //find interface number from ip
692  const char* ifacenhi_str = namesvc->ip2nicnhi(ip_target);
693  if(ifacenhi_str == 0) return 0; // ip_target does not belong to any host in this network
694  Nhi ifacenhi;
695  ifacenhi.convert((char*)ifacenhi_str, Nhi::NHI_INTERFACE);
696  int iface_num = ifacenhi.ids[ifacenhi.ids.size()-1];
697 
698  //find target veid from ip
699  int veid_target = namesvc->ip2veid(ip_target);
700 
701  if(veid_target < 0 || veid_target >= nr_ve)
702  {
703  //e.g. ip_target belongs to some simulated node, not emulated node
704  return 0;
705  }
706 
707  //deliver the packet to the target VE immediately
708  ppkt->ifid = iface_num;
709  ve[veid_target]->recvq.push_back(ppkt);
710  return 1;

789 void Net::injectEmuEvents()
790 {
791  // used to break tie for sending packet event with the same timestamp
792  // particularly need for events injected from the openvz emulation
793  int priority = 0;
794  IPADDR dst_ip;
795  IPADDR src_ip;
796 
797  if(isOpenVZEmu == true)
798  {
799  for (int i = 0; i < nr_ve; i++)
800  {
801  priority = 0;
802  while (!ve[i]->sendq.empty())
803  {
804  EmuPacket *ppkt = ve[i]->sendq.front();
805  ve[i]->sendq.pop_front();
806 
807  if(ppkt->len <= 0)
808  {
809  delete ppkt;
810  continue;
811  }
812 
813  Host* host = findOpenVZEmuHost(i);
814  if(host == NULL || host->isOpenVZEmu() == false)
815  {
816  delete ppkt;
817  continue;
818  }
819 
820  OpenVZEventSession* sess =
821  (OpenVZEventSession*)host->sessionForName(OPENVZ_EVENT_PROTOCOL_NAME);
822 
823  ethernet = (struct sniff_ethernet*)(ppkt->data);
824 
825  //printEthernetHeader();
826 
827  /* Do a couple of checks to see what packet type we have.*/
828  if(ntohs(ethernet->ether_type) == ETHERTYPE_IP)
829  {
830  NET_DUMP(printf("Net::injectEmuEvents(), Ethernet type hex:%x dec:%d is an IP packet\n",
831  ntohs(ethernet->ether_type), ntohs(ethernet->ether_type)));
832  dst_ip = getDstIPFromRealPacket(ppkt->data);
833  src_ip = getSrcIPFromRealPacket(ppkt->data);
834  }
835  else if(ntohs (ethernet->ether_type) == ETHERTYPE_ARP)
836  {
837  NET_DUMP(printf("Net::injectEmuEvents(), Ethernet type hex:%x dec:%d is an ARP packet\n",
838  ntohs(ethernet->ether_type), ntohs(ethernet->ether_type)));
839  if(process_ARP_packet(ppkt) == 0)
840  {
841  delete ppkt; //there is an error in the packet
842  }
843  continue;
844  }
845  else
846  {
847  printf("Net::injectEmuEvents(), Ethernet type %x not IP or ARP", ntohs(ethernet->ether_type));
848  delete ppkt;
849  continue;
850  }
851 
852  sess->injectEvent(dst_ip, src_ip, ppkt, priority);
853 
854  //we can edit VEHandle to record per VE in/out #packets
855  pkt_from_ve_counter++;
856 
857  priority++;
858  }
859  }
