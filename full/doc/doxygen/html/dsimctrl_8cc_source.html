725  maxadv = barrier - msg->int1;
726  newcycle = msg->int2;
727 
728  if (!newcycle) goto STEP2;
729 
730  /* step 1: lookahead pre_cycle for all VEs */
731  if (la_type) {
732  for (int i = 0; i < nve; i++)
733  if (ve[i]->la) ((Lookahead*)ve[i]->la)->pre_cycle(barrier);
734  }
735 
736  for (int i = 0; i < nve; i++) { done[i] = false; pktcount[i] = 0; }
737  ndone = 0;
738 
739 STEP2:
740  /* step 2: set the new barrier to all VEs */
741  if (newbarrier) {
742  if (debug) printf("new barrier for all VEs: %ld\n", barrier);
743  for (int i = 0; i < nve; i++)
744  if (vtime_setbarrier(ve[i]->id, &tvbarrier)) gotofail(-EINTR);
745  newbarrier = false;
746  }
747 
748 STEP3:
749  /* step 3: push due recv packets & due timers */
750  for (int i = 0; i < nve; i++) if (!done[i])
751  {
752  /* push recv packets */
753  deque<EmuPacket*> *pq1 = &(ve[i]->recvq);
754  while (!pq1->empty() && pq1->front()->timestamp <= ve[i]->clock) {
755  EmuPacket *ppkt = pq1->front();
756  pq1->pop_front();
757  ve[i]->idle = false;
758  if (ppkt->ifid >= 0 && ppkt->ifid < ve[i]->nif)
759  {
760  pcap_inject((pcap_t*)ve[i]->pcap[ppkt->ifid],
761  ppkt->data, ppkt->len);
762 #ifdef SLAVE_LOG
763  fprintf(pfslave, "%ld.%06ld\tve%d\trecv\t%d\n",
764  ve[i]->clock/1000000, ve[i]->clock%1000000,
765  ve[i]->id, ppkt->len);
766 #endif
767  if (debug) printf("VE%d recv packet, time=%ld size=%d\n",
768  ve[i]->id, ve[i]->clock, ppkt->len);
769  }
770  else {
771  fprintf(stderr, "simctrl warning: "
772  "ifid=%d, nif=%d, packet dropped\n",
773  ppkt->ifid, ve[i]->nif);
774  }
775  delete ppkt;
776  }
777  /* push due timers */
778  priority_queue<EmuTimer> *pq2 = &(ve[i]->timerq);
779  while (!pq2->empty() && pq2->top().due <= ve[i]->clock) {
780  EmuTimer timer = pq2->top();
781  pq2->pop();
782  ve[i]->idle = false;
783  if (timer.pid) vtime_firetimer(timer.pid);
784  if (debug) printf("VE%d timer due, time=%ld\n", ve[i]->id, timer.due);
785  if (timer.repeat) {
786  timer.due = ve[i]->clock + timer.repeat;
787  pq2->push(timer);
788  }
789  }
790  }
791 
792  /* step 4: release selected VEs */
793  if (debug) printf("maxadv=%ld barrier=%ld\n", maxadv, barrier);
794  again = false;
795  allidle = true;
796  for (int i = 0; i < nve; i++) {
797  rllist[i] = (!done[i])
798  && (!ve[i]->idle || ve[i]->idlecount > MAX_IDLE_COUNT)
799  && (ve[i]->clock < maxadv - min(ve[i]->offset, 0L));
800  if (!done[i] && ve[i]->clock < maxadv - min(ve[i]->offset, 0L)) again = true;
801  if (debug) printf("VE%d clock=%ld rllist=%d done=%d "
802  "idle=%d idlecount=%d offset=%ld\n",
803  ve[i]->id, ve[i]->clock, (int)rllist[i],
804  (int)done[i], (int)ve[i]->idle,
805  ve[i]->idlecount, ve[i]->offset);
806  if (rllist[i]) { allidle = false; ve[i]->idlecount = 0; }
807  }
808  if (allidle) goto STEP9;
809  if (vtime_releaseve(nve, rllist)) return -EINTR;
810 
811  /* step 5: wait until all VEs have stopped */
812  count = 0;
813  while (1)
814  {
815  int n;
816  bool ok = true;
817  if (vtime_getvestatus(&n, stlist)) return -EINTR;
818  if (n != nve) return -ESRCH;
819  for (int i = 0; i < nve; i++) {
820  if (0 && debug && count > 20) {
821  printf("VEs still not done, stlist =");
822  for (int i = 0; i < nve; i++) printf(" %d", stlist[i]);
823  printf("\n");
824  }
825  if (count > 100 && stlist[i] == VTIME_READY)
826  stlist[i] = VTIME_SUSPENDED;
827  if (stlist[i] != VTIME_SUSPENDED && stlist[i] != VTIME_IDLE) {
828  ok = false; /* some VEs still running */
829  sched_yield();
830  break;
831  }
832  }
833  if (ok) break;
834  count++;
835  }
836  if (debug) {
837  printf("VEs have stopped, stlist =");
838  for (int i = 0; i < nve; i++) printf(" %d", stlist[i]);
839  printf("\n");
840  }
841 
842  /* step 6: collect sent packets */
843  for (int i = 0; i < nve; i++) if (rllist[i])
844  {
845  for (int j = 0; j < ve[i]->nif; j++)
846  {
847  ve[i]->nowif = j;
848  while (pcap_dispatch((pcap_t*)ve[i]->pcap[j],
849  -1, got_packet, (u_char*)ve[i]) > 0)
850  stlist[i] = VTIME_SUSPENDED;
851  while (pktcount[i] < (int)ve[i]->sendq.size())
852  {
853  EmuPacket *ppkt = ve[i]->sendq[pktcount[i]++];
854 #ifdef SLAVE_LOG
855  fprintf(pfslave, "%ld.%06ld\tve%d\tsend\t%d\n",
856  ppkt->timestamp/1000000, ppkt->timestamp%1000000,
857  ve[i]->id, ppkt->len);
858 #endif
859  AsyncMessage *pktmsg= new AsyncMessage;
860  pktmsg->type = AsyncMessage::PACKET;
861  pktmsg->time = ppkt->timestamp;
862  pktmsg->int1 = ve[i]->id;
863  pktmsg->int2 = ve[i]->nowif;
864  pktmsg->resize(ppkt->len);
865  memcpy(pktmsg->payload, ppkt->data, ppkt->len);
866  ret = ((AsyncMessageHandler*)amh)->amh_send(pktmsg);
867  delete pktmsg;
868  if (ret) return ret;
869  }
870  }
871  }
872 
873  /* step 7: collect timers */
874  for (int i = 0; i < nve; i++) if (rllist[i])
875  {
876  int n;
877  if (vtime_gettraplist(ve[i]->id, &n, trlist)) return -EINTR;
878  for (int j = 0; j < n; j++)
879  if (trlist[j].type == VTIME_TRAPTIMER &&
880  trlist[j].data[0] > 0 && trlist[j].data[0] < 1000000)
881  {
882  EmuTimer timer;
883  timer.pid = trlist[j].pid;
884  timer.due = (trlist[j].data[0] +
885  trlist[j].time.tv_sec) * 1000000 +
886  trlist[j].data[1] + trlist[j].time.tv_usec;
887  timer.repeat = trlist[j].data[2]*1000000 +
888  trlist[j].data[3];
889  ve[i]->timerq.push(timer);
890  }
891  else if (trlist[j].type == VTIME_TRAPSLEEP &&
892  trlist[j].data[0] > 0 && trlist[j].data[0] < 1000000)
893  {
894  EmuTimer timer;
895  timer.pid = 0;
896  timer.due = trlist[j].data[0]*1000000 + trlist[j].data[1];
897  timer.repeat = 0;
898  ve[i]->timerq.push(timer);
899  }
900  /* release all traps */
901  for (int j = 0; j < n; j++) rtlist[j] = 1;
902  if (vtime_releasetrap(ve[i]->id, n, rtlist)) return -EINTR;
903  }
904 
905  /* step 8: update VE clocks */
906  for (int i = 0; i < nve; i++) if (rllist[i]) {
907  if (vtime_getclock(ve[i]->id, &tv)) return -EINTR;
908  ve[i]->lastclock = ve[i]->clock;
909  ve[i]->clock = tv.tv_sec*1000000 + tv.tv_usec;
910  if (debug) printf("VE%d clock=%ld lastc=%ld run=%ld\n", ve[i]->id,
911  ve[i]->clock, ve[i]->lastclock, ve[i]->clock-ve[i]->lastclock);
912  if (stlist[i] != VTIME_IDLE)
913  ve[i]->avgrun = (9 * ve[i]->avgrun +
914  (ve[i]->clock - ve[i]->lastclock)) / 10;
915  }
916 
917 STEP9:
918  /* step 9: adjust VEs virtual clocks */
919  alldone = true;
920  for (int i = 0; i < nve; i++) if (!done[i])
921  {
922  ltime_t adjbarrier = barrier - ve[i]->offset - ve[i]->avgrun/2;
923  if (debug) printf("VE%d offset=%ld avgrun=%ld adjbarrier=%ld\n",
924  ve[i]->id, ve[i]->offset, ve[i]->avgrun, adjbarrier);
925 
926  /* quick advance VE when idle */
927  if (stlist[i] == VTIME_IDLE)
928  {
929  ve[i]->idle = true;
930  ve[i]->idlecount++;
931  ltime_t next = min(maxadv, ve[i]->clock + VTIME_VTIMESLICE), ts;
932  if (!ve[i]->recvq.empty() &&
933  (ts = ve[i]->recvq.front()->timestamp) < adjbarrier)
934  if (next < 0 || ts < next) next = ts;
935  if (!ve[i]->timerq.empty() &&
936  (ts = ve[i]->timerq.top().due) < adjbarrier)
937  if (next < 0 || ts < next) next = ts;
938  if (next > ve[i]->clock) {
939  struct timeval tvnext;
940  tvnext.tv_sec = next / 1000000;
941  tvnext.tv_usec = next % 1000000;
942  if (vtime_setclock(ve[i]->id, &tvnext)) return -EINTR;
943  if (debug) printf("VE%d idle advance to %ld\n",
944  ve[i]->id, next);
945  ve[i]->offset = 0;
946  ve[i]->lastclock = ve[i]->clock = next;
947  }
948  }
949  else {
950  ve[i]->idle = false;
951  }
952 
953  /* VE has done current cycle */
954  if (ve[i]->clock >= adjbarrier)
955  {
956  if (debug) printf("VE%d reached barrier\n", ve[i]->id);
957  if (vtime_setclock(ve[i]->id, &tvbarrier)) return -EINTR;
958  if (stlist[i] != VTIME_IDLE) ve[i]->offset += ve[i]->clock - barrier;
959  else ve[i]->offset = 0;
960  ve[i]->lastclock = ve[i]->clock = barrier;
961  done[i] = true;
962  onedone = true;
963  ndone++;
964  }
965  else
966  alldone = false;
967  }
968 
969  if (debug && !debug_nopause) {
970  printf("press enter to continue\n");
971  char line[100];
972  fgets(line, 100, stdin);
973  if (strcmp(line, "exit") == 0) debug_nopause = true;
974  }
975 
976  if (!alldone && again) goto STEP3;
977  if (!alldone) goto EXIT;
978 
979  /* step 10: lookahead post_cycle for all VEs */
980  if (la_type) {
981  if (nthread == 1)
982  {
983  for (int i = 0; i < nve; i++) {
984  if (ve[i]->la) ((Lookahead*)ve[i]->la)->post_cycle(barrier);
985  else ve[i]->lookahead = 0;
986  }
987  }
988  else
989  {
990  /* use multi-thread to accelerate */
991  vector<pthread_t> tid(nthread);
992  void *status;
993  for (int i = 0; i < nthread; i++)
994  if (pthread_create(&tid[i], NULL, la_postcycle_thread, (void*)i))
995  gotofail(-ECHILD);
996  for (int i = 0; i < nthread; i++)
997  if (pthread_join(tid[i], &status)) gotofail(-ECHILD);
998  }
999 #ifdef TIME_STAT
1000  ltime_t minla = -1, totalla = 0;
1001  for (int i = 0; i < nve; i++) {
1002  if (minla < 0 || ve[i]->lookahead < minla) minla = ve[i]->lookahead;
1003  totalla += ve[i]->lookahead;
1004  }
1005  if (!timestat_counter) timestat_update(avgminla, minla);
1006  if (!timestat_counter) timestat_update(avgla, totalla / nve);
1007 #endif
1008  }
1009 
1010  /* empty sendq */
1011  for (int i = 0; i < nve; i++)
1012  while (!ve[i]->sendq.empty()) {
1013  EmuPacket *ppkt = ve[i]->sendq.front();
1014  ve[i]->sendq.pop_front();
1015  delete ppkt;
1016  }
1017 
1018 #ifdef TIME_STAT
1019  timestat_counter = (timestat_counter + 1) % timestat_interval;
1020  if (!timestat_counter)
1021  {
1022  gettimeofday(&tstv2, NULL);
1023  int elapse = VTIME_CLOCKDIFFU(tstv2, lastprint);
1024  if (elapse > 1000000)
1025  {
1026  double speed = (double)(barrier - lastvtime) / elapse;
1027  lastprint = tstv2;
1028  lastvtime = barrier;
1029  fprintf(stderr, "vtime=%d.%03d speed=%.3f",
1030  (int)(barrier/1000000), (int)(barrier%1000000/1000), speed);
1031  if (la_type) {
1032  double err = 0.0;
1033  for (int i = 0; i < nve; i++)
1034  err += ((Lookahead*)ve[i]->la)->get_error();
1035  err /= nve;
1036  laerr = (int)(err * 10000);
1037  fprintf(stderr, " err=%.0f%% avgla=%.0f avgminla=%.0f",
1038  err*100, avgla, avgminla);
1039  }
1040  fprintf(stderr, "\n");
1041  }
1042  }
1043 #endif
1044 
1045 EXIT:
1046  if (debug) printf("----- cycle done -----\n");
1047 
1048  /* reply message */
1049  reply = new AsyncMessage;
1050  reply->type = AsyncMessage::SUCC;
1051  reply->int1 = ndone;
1052  reply->int2 = laerr;
1053  reply->resize(nve * sizeof(ltime_t));
1054  p = (ltime_t*) reply->payload;
1055  for (int i = 0; i < nve; i++, p++)
1056  *p = alldone ? ve[vemap[i]]->lookahead : ve[vemap[i]]->clock;
1057  ret = ((AsyncMessageHandler*)amh)->amh_send(reply);
1058  delete reply;
1059  return ret;
1060 
1061 FAIL:
1062  reply = new AsyncMessage;
1063  reply->type = AsyncMessage::FAIL;
1064  reply->int1 = reason;
1065  ret = ((AsyncMessageHandler*)amh)->amh_send(reply);
1066  delete reply;
1067  return ret;
1068 }
