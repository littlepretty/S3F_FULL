208 void TCPSession::appl_close()
209 {
210  assert(sndwnd);
211  if(sndwnd->empty() && sndwnd->dataInBuffer() == 0)
212  {
213  // if there is nothing to send
214  switch(state) {
215  case TCP_STATE_ESTABLISHED:
216  // go to state TCP_STATE_FIN_WAIT_1
217  init_state_fin_wait_1();
218  return;
219 
220  case TCP_STATE_CLOSE_WAIT:
221  // send out FIN
222  send_data(sndwnd->firstUnused(), 0, TCPMessage::TCP_FLAG_FIN,
223  rcvwnd->expect(), true, true);
224  sndwnd->setFIN(true);
225  // go to state TCP_STATE_LAST_ACK
226  init_state_last_ack();
227  return;
228 
229  case TCP_STATE_TIME_WAIT:
230  wake_app(SocketSignal::CONN_CLOSED);
231  return;
232  }
233  }
234  else
235  {
236  clear_app_state(SocketSignal::OK_TO_CLOSE);
237  close_issued = true;
238  }
239 }
