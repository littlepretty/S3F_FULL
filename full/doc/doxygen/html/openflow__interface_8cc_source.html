300 int Stats::PortStatsDump (OpenFlowSwitchSession* swtch, PortStatsState *s, ofpbuf *buffer)
301 {
302  ofp_port_stats *ops;
303  uint32_t port;
304 
305  // port stats are different depending on whether port is physical or virtual
306  for (size_t i = 0; i < s->num_ports; i++)
307  {
308  port = ntohl (s->ports[i]);
309  // physical port?
310  if (port <= OFPP_MAX)
311  {
312  Port p = swtch->GetSwitchPort (port);
313 
314  if (p.netdev == 0)
315  {
316  continue;
317  }
318 
319  ops = (ofp_port_stats*)ofpbuf_put_zeros (buffer, sizeof *ops);
320  ops->port_no = htonl (swtch->GetSwitchPortIndex (p));
321  ops->rx_packets = htonll (p.rx_packets);
322  ops->tx_packets = htonll (p.tx_packets);
323  ops->rx_bytes = htonll (p.rx_bytes);
324  ops->tx_bytes = htonll (p.tx_bytes);
325  ops->rx_dropped = htonll (-1);
326  ops->tx_dropped = htonll (p.tx_dropped);
327  ops->rx_errors = htonll (-1);
328  ops->tx_errors = htonll (-1);
329  ops->rx_frame_err = htonll (-1);
330  ops->rx_over_err = htonll (-1);
331  ops->rx_crc_err = htonll (-1);
332  ops->collisions = htonll (-1);
333  ops->mpls_ttl0_dropped = htonll (p.mpls_ttl0_dropped);
334  ops++;
335  }
336  else if (port >= OFPP_VP_START && port <= OFPP_VP_END) // virtual port?
337  {
338  // lookup the virtual port
339  vport_table_t vt = swtch->GetVPortTable ();
340  vport_table_entry *vpe = vport_table_lookup (&vt, port);
341  if (vpe == 0)
342  {
343  OF_IFACE_DUMP (cout << "vport entry not found!" << endl);
344  continue;
345  }
346  // only tx_packets and tx_bytes are really relevant for virtual ports
347  ops = (ofp_port_stats*)ofpbuf_put_zeros (buffer, sizeof *ops);
348  ops->port_no = htonl (vpe->vport);
349  ops->rx_packets = htonll (-1);
350  ops->tx_packets = htonll (vpe->packet_count);
351  ops->rx_bytes = htonll (-1);
352  ops->tx_bytes = htonll (vpe->byte_count);
353  ops->rx_dropped = htonll (-1);
354  ops->tx_dropped = htonll (-1);
355  ops->rx_errors = htonll (-1);
356  ops->tx_errors = htonll (-1);
357  ops->rx_frame_err = htonll (-1);
358  ops->rx_over_err = htonll (-1);
359  ops->rx_crc_err = htonll (-1);
360  ops->collisions = htonll (-1);
361  ops->mpls_ttl0_dropped = htonll (-1);
362  ops++;
363  }
364  }
365  return 0;
366 }

625  /* Every output action needs a separate clone of 'buffer', but the common
626  * case is just a single output action, so that doing a clone and then
627  * freeing the original buffer is wasteful. So the following code is
628  * slightly obscure just to avoid that. */
629  int prev_port;
630  size_t max_len = 0; // Initialze to make compiler happy
631  uint16_t in_port = key->flow.in_port; // ntohs(key->flow.in_port);
632 
633  uint8_t *p = (uint8_t *)actions;
634 
635  prev_port = -1;
636 
637  if (actions_len == 0)
638  {
639  OF_IFACE_DUMP (cout << "No actions set to this flow. Dropping packet." << endl);
640  return;
641  }
642 
643  /* The action list was already validated, so we can be a bit looser
644  * in our sanity-checking. */
645  while (actions_len > 0)
646  {
647  ofp_action_header *ah = (ofp_action_header *)p;
648  size_t len = htons (ah->len);
649 
650  if (prev_port != -1)
651  {
652  swtch->DoOutput (packet_uid, in_port, max_len, prev_port, ignore_no_fwd);
653  prev_port = -1;
654  }
655 
656  if (ah->type == htons (OFPAT_OUTPUT))
657  {
658  ofp_action_output *oa = (ofp_action_output *)p;
659 
660  // port is now 32-bits
661  prev_port = oa->port; // ntohl(oa->port);
662  max_len = ntohs (oa->max_len);
663  }
664  else
665  {
666  uint16_t type = ntohs (ah->type);
667  if (Action::IsValidType ((ofp_action_type)type)) // Execute a built-in OpenFlow action against 'buffer'.
668  {
669  Action::Execute ((ofp_action_type)type, buffer, key, ah);
670  }
671  else if (type == OFPAT_VENDOR)
672  {
673  ExecuteVendor (buffer, key, ah);
674  }
675  }
676 
677  p += len;
678  actions_len -= len;
679  }
680 
681  if (prev_port != -1)
682  {
683  swtch->DoOutput (packet_uid, in_port, max_len, prev_port, ignore_no_fwd);
684  }
685 }

687 uint16_t ValidateActions (const sw_flow_key *key, const ofp_action_header *actions, size_t actions_len)
688 {
689  uint8_t *p = (uint8_t *)actions;
690  int err;
691 
692  while (actions_len >= sizeof(ofp_action_header))
693  {
694  ofp_action_header *ah = (ofp_action_header *)p;
695  size_t len = ntohs (ah->len);
696  uint16_t type;
697 
698  /* Make there's enough remaining data for the specified length
699  * and that the action length is a multiple of 64 bits. */
700  if ((actions_len < len) || (len % 8) != 0)
701  {
702  return OFPBAC_BAD_LEN;
703  }
704 
705  type = ntohs (ah->type);
706  if (Action::IsValidType ((ofp_action_type)type)) // Validate built-in OpenFlow actions.
707  {
708  err = Action::Validate ((ofp_action_type)type, len, key, ah);
709  if (err != ACT_VALIDATION_OK)
710  {
711  return err;
712  }
713  }
714  else if (type == OFPAT_VENDOR)
715  {
716  err = ValidateVendor (key, ah, len);
717  if (err != ACT_VALIDATION_OK)
718  {
719  return err;
720  }
721  }
722  else
723  {
724  return OFPBAC_BAD_TYPE;
725  }
726 
727  p += len;
728  actions_len -= len;
729  }
730 
731  // Check if there's any trailing garbage.
732  if (actions_len != 0)
733  {
734  return OFPBAC_BAD_LEN;
735  }
736 
737  return ACT_VALIDATION_OK;
738 }

834 void ExecuteVendor (ofpbuf *buffer, const sw_flow_key *key, const ofp_action_header *ah)
835 {
836  ofp_action_vendor_header *avh = (ofp_action_vendor_header *)ah;
837 
838  switch (ntohl (avh->vendor))
839  {
840  case NX_VENDOR_ID:

849  default:
850  // This should not be possible due to prior validation.
851  OF_IFACE_DUMP (cout << "attempt to execute action with unknown vendor: " << ntohl (avh->vendor) << endl);
852  break;
853  }
