609 int NBSocketMaster::send(int sock, uint32 length, byte* msg, NBSocketContinuation *caller)
610 {
611  assert(caller);
612 
613  Socket* mysocket = 0;
614  S3FNET_MAP(int, Socket*)::iterator iter = bound_socks.find(sock);
615  if(iter != bound_socks.end()) mysocket = (*iter).second;
616  if(!mysocket || !mysocket->session || !mysocket->session->connected())
617  {
618  caller->failure();
619  return EGENERIC;
620  }
621 
622  if(mysocket->getResume() & Socket::RESUME_SEND1)
623  {
624  int res = block_till(sock, Socket::OK_TO_SEND|Socket::CONN_RESET, false, caller);
625  if(res == ESUCCESS) {
626  res = send1(sock,caller);
627  }
628  mysocket->setResume(Socket::RESUME_NONE);
629  return res;
630  }
631 
632  SOCK_DUMP(printf("[host=\"%s\"] %ld: send(), socket=%d (state=%08x), length=%u (%s).\n",
633  inHost()->nhi.toString(), inHost()->now(), sock, mysocket->state, length, msg?"real":"virtual"));
634 
635  // initiate the send
636  mysocket->bytes_completed = mysocket->session->send(length, msg);
637  if(mysocket->bytes_completed < 0)
638  {
639  caller->failure();
640  return EGENERIC;
641  }
642 
643  // if initially the message has been sent completely in all, no more
644  // waiting is necessary; otherwise, one must wait for the OK_TO_SEND
645  // (or CONN_RESET) signal to get the true number of bytes that have
646  // completed the transmission; we catch RESUME_SEND1 above
647  if(mysocket->bytes_completed < length)
648  {
649  int res = block_till(sock, Socket::OK_TO_SEND|Socket::CONN_RESET, false, caller);
650  if(res == ESUCCESS) {
651  res = send1(sock,caller);
652  } else {
653  mysocket->setResume(mysocket->getResume() | Socket::RESUME_SEND1);
654  }
655  return res;
656  }
657  else
658  {
659  SOCK_DUMP(printf("[host=\"%s\"] %ld: send(), socket=%d (state=%08x), sent all %u bytes, immediately.\n",
660  inHost()->nhi.toString(), inHost()->now(), sock, mysocket->state, length));
661  caller->retval = length;
662  caller->success();
663  return ESUCCESS;
664  }

808 int NBSocketMaster::close(int sock, NBSocketContinuation *caller)
809 {
810  assert(caller);
811 
812  Socket* mysocket = 0;
813  S3FNET_MAP(int, Socket*)::iterator iter = bound_socks.find(sock);
814  if(iter != bound_socks.end()) mysocket = (*iter).second;
815  if(!mysocket)
816  {
817  caller->failure();
818  return EGENERIC;
819  }
820 
821  if(mysocket->getResume() & Socket::RESUME_CLOSE1) {
822  int res = block_till(sock, Socket::CONN_RESET|Socket::CONN_CLOSED|Socket::SESSION_RELEASED, false, caller);
823  if(res == ESUCCESS) {
824  mysocket->setResume(Socket::RESUME_NONE);
825  res = close1(sock,caller);
826  }
827  return res;
828  }
829 
830  SOCK_DUMP(printf("[host=\"%s\"] %ld: close(), socket=%d (state=%08x).\n",
831  inHost()->nhi.toString(), inHost()->now(), sock, mysocket->state));
832 
833  if(mysocket->session && mysocket->session->connected())
834  {
835  // proceed to disconnect
836  mysocket->session->disconnect();
837 
838  int res = block_till(sock, Socket::CONN_RESET|Socket::CONN_CLOSED|Socket::SESSION_RELEASED, false, caller);
839  if(res == ESUCCESS) {
840  res = close1(sock,caller);
841  } else {
842  mysocket->setResume(Socket::RESUME_CLOSE1);
843  }
844  return res;
845  }
846  else
847  {
848  SOCK_DUMP(printf("[host=\"%s\"] %ld: close(), socket=%d (state=%08x), not connected (no blocking).\n",
849  inHost()->nhi.toString(), inHost()->now(), sock, mysocket->state));
850 
851  mysocket->reinit();
852  if(mysocket->is_active()) mysocket->die_slowly();
853  else
854  {
855  // the destructor will call release method---which will detach
856  // the socket session from the socket, remove the socket session
857  // from the session master, and then reclaim the socket
858  // session---before the socket is reclaimed
859  mysocket->release();
860  bound_socks.erase(sock);
861  }
862  caller->success();
863  return ESUCCESS;
864  }
865 }
