553  // create the fast timer
554  Host* owner_host = inHost();
555  fast_timer_callback_proc = new Process( (Entity *)owner_host,
556  (void (s3f::Entity::*)(s3f::Activation))&TCPMaster::fast_timer_callback);
557  // start the fast timer
558  fast_timer_ac = new ProtocolCallbackActivation(this);
559  Activation ac (fast_timer_ac);
560  double time = inHost()->t2d(getNow(), 0);
561  double t = calc_fast_timeout(time);
562  HandleCode h = owner_host->waitFor( fast_timer_callback_proc, ac, inHost()->d2t(t, 0), owner_host->tie_breaking_seed);
563  }
564  fast_timeout_sessions.insert(session);
565 }

580 {
581  delete_defunct_sessions();
582 
583  // go through each tcp session that requires fast timer; to prevent
584  // changes during iteration, we copy the list to an array.
585  int mysize;
586  if((mysize = fast_timeout_sessions.size()) > 0)
587  {
588  TCPSession** mylist = new TCPSession*[mysize];
589  assert(mylist);
590  int idx = 0;
591  for(S3FNET_SET(TCPSession*)::iterator iter = fast_timeout_sessions.begin(); iter != fast_timeout_sessions.end(); iter++, idx++)
592  mylist[idx] = (*iter);
593  for(idx = 0; idx < mysize; idx++)
594  mylist[idx]->fastTimeoutHandling();
595  delete[] mylist;
596  }

599  double time = inHost()->t2d(getNow(), 0);
600  double t = calc_fast_timeout(time);
