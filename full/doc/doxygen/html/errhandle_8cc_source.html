27 static void default_errhandler(int syserr, const char* fmt, va_list ap)
28 {
29  // Stack memory instead of heap memory is desired here as the error
30  // could come from the memory system. However, this could lead to
31  // buffer overflow problem if not dealt with carefully. We must
32  // truncate the error message if there's a buffer overflow.
33  char buf[S3FNET_ERRHANDLE_PRTFMT_BUFSIZ];
34 
35  if(fmt) {
36  // We must save the error number in case it's overwritten by
37  // other system calls in this routine.
38  int errno_save = errno;
39 
40  // Write the message to the buffer.
41  vsnprintf(buf, S3FNET_ERRHANDLE_PRTFMT_BUFSIZ, fmt, ap);
42  buf[S3FNET_ERRHANDLE_PRTFMT_BUFSIZ-1] = 0; // safety
43  int offset = strlen(buf);
44 
45  // Append system error messages if this is related to a system
46  // call. Applicable only if there's enough buffer space.
47  if(syserr && offset < S3FNET_ERRHANDLE_PRTFMT_BUFSIZ-1) {
48  snprintf(buf+offset, S3FNET_ERRHANDLE_PRTFMT_BUFSIZ-1-offset,
49  ": %s", strerror(errno_save));
50  buf[S3FNET_ERRHANDLE_PRTFMT_BUFSIZ-1] = 0; // safety
51  offset = strlen(buf);
52  }
53 
54  // We add a newline at the end of the message. So, no need to add
55  // a newline in the format string!
56  if(offset < S3FNET_ERRHANDLE_PRTFMT_BUFSIZ-1)
57  strcat(buf, "\n");
58 
59  // Flush all file streams, including stdout, so that regular
60  // messages are printed out before the error message.
61  fflush(0);
62 
63  // Print out error message to stderr.
64  fputs(buf, stderr);
65 
66  // Flush one more time.
67  fflush(0);
68  }
69 }

71 // Global variables must be dealt with caution in s3fnet, since they may
72 // not be shared by all processes in a multi-processor environment.
73 // Actually there are two possible cases. In the first case, there is
74 // one instance of a global variable defined for each process. The
75 // global data segment is not shared, but copied instead at the moment
76 // of spawning child processes. In the second case, there is exactly
77 // one variable shared by all processors, unprotected from
78 // simultaneous access. We initialize the global error handler to be
79 // the default one. We must make sure that, if we want to replace it
80 // with a user-defined error handler, it must be done BEFORE the child
81 // processes are created. Another consequence is that the error
82 // handler may not be available at the C global initialization
83 // phase. Also it is possible some global variables may be initialized
84 // before this assignment happens. In any case we assume the handler
85 // is NULL initially. But if it's not, we'll be in trouble!
86 static ErrorHandler errhandler = default_errhandler;
